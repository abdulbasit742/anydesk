 Codex / Claude Code handoff. Direct continuation of Desktop Packs 15 → 19. Everything so far assumes a
 device already has a trustStatus . This slice builds the moment that status is first established: a stable, non-
 sensitive device fingerprint, a viewer-facing trust prompt on first connect, and a trust-on-first-use (TOFU)
 decision flow that detects fingerprint CHANGES on a known device and downgrades to untrusted until re-
 approved. Build order: shared fingerprint/trust contracts → fingerprint builder → trust decision store → trust
 prompt UI → session wiring → tests.
 Scope: establish + verify device identity before trust is granted, and catch identity changes on return visits.
 Fail-closed: unknown or changed fingerprint => untrusted (which Pack 15 already uses to deny remote
 input). No device secrets, IPs, or hardware serials in the fingerprint or logs. Host stays source of truth.
 No native executor.

FILE: packages/shared/src/permissions/deviceFingerprint.ts
   /**
       * Non-sensitive device fingerprint. Built ONLY from coarse, non-identifying
       * signals (platform family, app version, screen bucket, locale, a per-install
       * random id the user can reset). NO IPs, MAC/serials, usernames, or hardware ids.
       * The fingerprint is a stable hash used to recognize a returning install and to
       * detect material changes — not to uniquely track hardware.
       */
   export interface FingerprintSignals {
     /** "mac" | "win" | "linux" — family only, not full OS build. */
        platformFamily: string;
        /** App semver, e.g. "1.4.0". */
        appVersion: string;
        /** Coarse screen bucket, e.g. "1920x1080" rounded — not exact geometry. */
        screenBucket: string;
        /** BCP-47 language, e.g. "en-US". */
        locale: string;
        /** Per-install random id (persisted locally, user-resettable). Not hardware-derived. */
        installId: string;
   }

   export interface DeviceFingerprint {
        /** Stable hash (hex) of the canonical signals. */
        hash: string;
        /** The signals that produced the hash (for change diffing + display). */
        signals: FingerprintSignals;
        /** ISO of when it was computed. */
        computedAt: string;
   }


   /** Canonical string for hashing: order-stable, lowercased keys. */
   export function canonicalizeSignals(s: FingerprintSignals): string {
        return [
          `platform=${s.platformFamily}`,
          `app=${s.appVersion}`,
          `screen=${s.screenBucket}`,
          `locale=${s.locale}`,
          `install=${s.installId}`,
        ].join('&');
   }


   /**
       * Which signal changes are "material" (should trigger re-trust). installId change
       * is the strongest signal (fresh install / reset). platformFamily change is also
       * material. appVersion/screen/locale drift alone is NOT material on its own.
       */
   export function materialSignalChanges(
     prev: FingerprintSignals,
     next: FingerprintSignals,
   ): string[] {
        const changes: string[] = [];
        if (prev.installId !== next.installId) changes.push('installId');
        if (prev.platformFamily !== next.platformFamily) changes.push('platformFamily');
        return changes;
   }


   /** True if the change set warrants dropping back to untrusted. */
   export function isMaterialChange(prev: FingerprintSignals, next: FingerprintSignals): boolean {
     return materialSignalChanges(prev, next).length > 0;
   }



FILE: packages/shared/src/permissions/trustDecision.ts
   import type { DeviceTrustStatus } from './devicePolicy';
   import { isMaterialChange, type DeviceFingerprint } from './deviceFingerprint';

   /**
    * Trust-on-first-use (TOFU) decision logic. Pure + fail-closed:
       *       - first time we see a device => 'prompt' (ask the operator; status untrusted until approved)
       *       - known device, same fingerprint => keep persisted status
       *       - known device, MATERIAL fingerprint change => 'prompt' + force untrusted
       *       - blocked is sticky: a blocked device is never auto-trusted by TOFU
    */
   export type TrustGateOutcome = 'trusted' | 'untrusted' | 'blocked' | 'prompt';

   export interface TrustEvaluationInput {
           /** Persisted trust status from the API, or null if the device is unknown. */
           persistedStatus: DeviceTrustStatus | null;
           /** Fingerprint stored alongside the persisted status, or null if none. */
           knownFingerprint: DeviceFingerprint | null;
           /** Freshly computed fingerprint for this connection. */
           currentFingerprint: DeviceFingerprint;
   }


   export interface TrustEvaluation {
     outcome: TrustGateOutcome;
           /** Effective trust status to feed into the policy snapshot (fail-closed). */
           effectiveStatus: DeviceTrustStatus;
           /** Why, for UI + audit (non-sensitive). */
           reason: 'first_use' | 'unchanged' | 'fingerprint_changed' | 'blocked';
   }


   export function evaluateTrust(input: TrustEvaluationInput): TrustEvaluation {
     const { persistedStatus, knownFingerprint, currentFingerprint } = input;


           // Blocked is sticky and overrides everything.
           if (persistedStatus === 'blocked') {
             return { outcome: 'blocked', effectiveStatus: 'blocked', reason: 'blocked' };
           }

           // Unknown device OR no stored fingerprint => first use => prompt, untrusted now.
           if (persistedStatus === null || knownFingerprint === null) {
               return { outcome: 'prompt', effectiveStatus: 'untrusted', reason: 'first_use' };
           }


           // Known device: detect a material fingerprint change => re-prompt, untrusted.
           if (isMaterialChange(knownFingerprint.signals, currentFingerprint.signals)) {
             return { outcome: 'prompt', effectiveStatus: 'untrusted', reason: 'fingerprint_changed' };
           }
        // Stable + known: keep the persisted status (trusted or untrusted).
        return {
          outcome: persistedStatus === 'trusted' ? 'trusted' : 'untrusted',
            effectiveStatus: persistedStatus,
            reason: 'unchanged',
        };
   }


   /** The operator's response to a trust prompt. */
   export type TrustPromptResponse = 'trust' | 'keep_untrusted' | 'block';


   /** Map a prompt response to the resulting persisted status. */
   export function statusFromPromptResponse(r: TrustPromptResponse): DeviceTrustStatus {
        switch (r) {
          case 'trust': return 'trusted';
            case 'block': return 'blocked';
            case 'keep_untrusted':
            default: return 'untrusted';
        }
   }

   import {
        canonicalizeSignals,
        type DeviceFingerprint,
     type FingerprintSignals,
   } from '@remotedesk/shared';


   /**
       * Builds a non-sensitive device fingerprint from coarse renderer-available signals.
       * Deliberately avoids IPs, MAC/serials, usernames, GPU strings, etc. The installId
       * is a per-install random value persisted in localStorage and user-resettable.
       */
   const INSTALL_ID_KEY = 'remotedesk.installId';


   export async function buildFingerprint(appVersion: string): Promise<DeviceFingerprint> {
     const signals: FingerprintSignals = {
            platformFamily: detectPlatformFamily(),
            appVersion,
        screenBucket: screenBucket(),
        locale: typeof navigator !== 'undefined' ? navigator.language : 'unknown',
        installId: getOrCreateInstallId(),
    };
    return {
        hash: await sha256Hex(canonicalizeSignals(signals)),
        signals,
        computedAt: new Date().toISOString(),
    };
}


/** Reset the per-install id (e.g. a "forget this device" action). */
export function resetInstallId(): void {
    try {
      localStorage.setItem(INSTALL_ID_KEY, cryptoRandomId());
    } catch {
      /* storage unavailable: next build falls back to an ephemeral id */
    }
}

function detectPlatformFamily(): string {
    const p = typeof navigator !== 'undefined' ? navigator.platform.toLowerCase() : '';
    if (p.includes('mac')) return 'mac';
    if (p.includes('win')) return 'win';
    if (p.includes('linux')) return 'linux';
    return 'unknown';
}


/** Coarse screen bucket: round to the nearest 160px so minor changes don't churn. */
function screenBucket(): string {
  if (typeof screen === 'undefined') return 'unknown';
    const round = (n: number) => Math.round(n / 160) * 160;
    return `${round(screen.width)}x${round(screen.height)}`;
}


function getOrCreateInstallId(): string {
  try {
        const existing = localStorage.getItem(INSTALL_ID_KEY);
        if (existing) return existing;
        const id = cryptoRandomId();
        localStorage.setItem(INSTALL_ID_KEY, id);
      return id;
    } catch {
        return cryptoRandomId(); // ephemeral fallback
    }
}

function cryptoRandomId(): string {
  const bytes = new Uint8Array(16);
    crypto.getRandomValues(bytes);
    return Array.from(bytes, (b) => b.toString(16).padStart(2, '0')).join('');
}

async function sha256Hex(input: string): Promise<string> {
  const data = new TextEncoder().encode(input);
    const digest = await crypto.subtle.digest('SHA-256', data);
        return Array.from(new Uint8Array(digest), (b) => b.toString(16).padStart(2, '0')).join('');
   }

FILE: apps/desktop/src/renderer/src/services/trustDecisionStore.ts
   import {
        evaluateTrust,
        statusFromPromptResponse,
        type DeviceFingerprint,
        type DeviceTrustStatus,
        type TrustEvaluation,
        type TrustPromptResponse,
   } from '@remotedesk/shared';


   /**
    * Orchestrates the TOFU flow on the desktop. Evaluates trust from the persisted
       * status + known fingerprint vs the freshly built one, exposes whether a prompt
       * is needed, and persists the operator's decision back to the API. Fail-closed:
       * the effectiveStatus is always untrusted (or blocked) until a prompt is resolved.
       */
   export interface TrustPersistence {
     /** Persist the operator's trust decision + current fingerprint to the API. */
        persist: (deviceId: string, status: DeviceTrustStatus, fingerprint: DeviceFingerprint) => Promise<void>;
   }


   export class TrustDecisionStore {
        private evaluation: TrustEvaluation | null = null;
        private listeners = new Set<(e: TrustEvaluation) => void>();


        constructor(private readonly persistence: TrustPersistence) {}

        /** Evaluate trust for this connection. Returns the evaluation (also cached). */
        evaluate(input: {
          persistedStatus: DeviceTrustStatus | null;
            knownFingerprint: DeviceFingerprint | null;
            currentFingerprint: DeviceFingerprint;
        }): TrustEvaluation {
          this.evaluation = evaluateTrust(input);
            this.emit();
            return this.evaluation;
        }


        current(): TrustEvaluation | null {
          return this.evaluation;
        }


        /** Does the UI need to show a trust prompt right now? */
        needsPrompt(): boolean {
            return this.evaluation?.outcome === 'prompt';
        }

        /** Effective trust status to feed the policy snapshot (fail-closed default). */
        effectiveStatus(): DeviceTrustStatus {
            return this.evaluation?.effectiveStatus ?? 'untrusted';
        }


        /** Resolve a pending prompt: persist + update the cached evaluation. */
        async resolvePrompt(
            deviceId: string,
            response: TrustPromptResponse,
          currentFingerprint: DeviceFingerprint,
        ): Promise<DeviceTrustStatus> {
            const status = statusFromPromptResponse(response);
            try {
              await this.persistence.persist(deviceId, status, currentFingerprint);
            } catch {
              // Persistence failed: stay fail-closed (do not optimistically trust).
              return this.effectiveStatus();
            }
            this.evaluation = {
              outcome: status === 'trusted' ? 'trusted' : status === 'blocked' ? 'blocked' : 'untrusted',
              effectiveStatus: status,
              reason: 'unchanged',
            };
            this.emit();
            return status;
        }


        subscribe(fn: (e: TrustEvaluation) => void): () => void {
          this.listeners.add(fn);
            return () => this.listeners.delete(fn);
        }


        private emit(): void {
            if (this.evaluation) for (const fn of this.listeners) fn(this.evaluation);
        }
   }

   import React from 'react';
   import type { TrustEvaluation, TrustPromptResponse } from '@remotedesk/shared';


   /**
    * Trust-on-first-use prompt. Shown when the trust evaluation outcome is 'prompt'
    * (first connect or a material fingerprint change). Until resolved, the device is
    * untrusted, so Pack 15 already denies remote input. Screen viewing is unaffected.
    */
   export function TrustPrompt({
     evaluation,
     onRespond,
   }: {
        evaluation: TrustEvaluation;
        onRespond: (r: TrustPromptResponse) => void;
   }) {
     if (evaluation.outcome !== 'prompt') return null;


        const headline =
         evaluation.reason === 'fingerprint_changed'
           ? 'This device looks different from last time'
            : 'New device connecting';
        const sub =
         evaluation.reason === 'fingerprint_changed'
           ? 'Its identity changed since you last trusted it. Remote input stays disabled until you decide.'
           : 'You have not trusted this device before. Remote input stays disabled until you decide.';


        return (
          <div className="trust-prompt" role="dialog" aria-modal="false">
           <h3 className="trust-prompt__title">{headline}</h3>
           <p className="trust-prompt__sub">{sub}</p>
           <div className="trust-prompt__actions">
             <button type="button" className="btn btn--primary" onClick={() => onRespond('trust')}>Trust this device</button>
           <button type="button" className="btn btn--ghost" onClick={() => onRespond('keep_untrusted')}>Keep untrusted (view
   only)</button>
             <button type="button" className="btn btn--danger" onClick={() => onRespond('block')}>Block</button>
           </div>
         </div>
       );
  }

  import React from 'react';
  import { materialSignalChanges, type DeviceFingerprint } from '@remotedesk/shared';


  /**
   * Compact banner noting which signals changed since the last trusted fingerprint.
      * Non-sensitive (signal names only, never values like installId contents).
      */
  export function FingerprintChangeBanner({
    known,
    current,
  }: {
       known: DeviceFingerprint | null;
       current: DeviceFingerprint | null;
  }) {
    if (!known || !current) return null;
       const changes = materialSignalChanges(known.signals, current.signals);
       if (changes.length === 0) return null;
       return (
         <div className="fingerprint-change-banner" role="status">
           Device identity changed: {changes.join(', ')}. Re-approval required before remote input.
         </div>
       );
  }
