RemoteDesk · Pack 22 (Desktop) — Session Summary +
Verifiable Consent Receipt at Disconnect (real code)
 Codex / Claude Code handoff. Direct continuation of Desktop Packs 15 → 21. The permission lifecycle is
 complete; this slice produces the closing artifact: when a session ends, build a content-free session summary
 + a consent receipt the user can keep as proof of exactly what was shared, for how long, under what
 consent, and what control happened. The receipt is verifiable offline via a hash chain over the session's own
 audit timeline (Pack 17/18). Build order: shared receipt contract → summary builder → receipt builder +
 offline verifier → disconnect wiring → UI → tests.
 Scope: a tamper-evident, non-sensitive record of the session, derived entirely from data already collected
 (timeline, permissions, grant). No payloads, keystrokes, file contents, IPs, or secrets in the receipt or
 summary. Verifiable without the server. Host stays source of truth. No native executor.

  import type { SessionAuditEvent } from './sessionAuditEvents';
  import type { SessionPermissionSet } from './devicePolicy';

  /**
   * A content-free summary of a finished session, derived from data already
      * collected during the session (audit timeline + final permission set). NO
      * payloads, keystrokes, file contents, IPs, or secrets — only counts, durations,
      * and which capabilities were ever active.
      */
  export interface SessionSummary {
    deviceId: string;
       startedAt: string;
       endedAt: string;
       durationSeconds: number;
       /** Capabilities that were enabled at any point during the session. */
       capabilitiesUsed: {
         remoteInput: boolean;
         clipboardSync: boolean;
         fileTransfer: boolean;
       };
       /** Whether the session ran unattended (auto-accepted under a grant). */
       unattended: boolean;
       /** Aggregate counts from the audit timeline. */
       counts: {
         inputBlocked: number;
         inputRateLimited: number;
         emergencyStops: number;
         controlGrants: number;
         controlRevokes: number;
       };
       /** How the session ended. */
       endReason: 'user_disconnect' | 'expired' | 'reconnect_failed' | 'unknown';
  }


  /** Count events of a given type in the timeline. */
  export function countEvents(events: SessionAuditEvent[], type: string): number {
    return events.filter((e) => e.type === type).length;
  }

  /**
   * Build a summary from the audit timeline + bounds. Pure: capability usage is
      * inferred from control/feature events + the peak permission set seen.
      */
  export function buildSessionSummary(input: {
    deviceId: string;
       startedAt: string;
       endedAt: string;
       events: SessionAuditEvent[];
       peakPermissions: Pick<SessionPermissionSet, 'remoteInput' | 'clipboardSync' | 'fileTransfer'> | null;
       unattended: boolean;
       endReason: SessionSummary['endReason'];
  }): SessionSummary {
    const durationSeconds = Math.max(
         0,
         Math.round((Date.parse(input.endedAt) - Date.parse(input.startedAt)) / 1000),
        );
        return {
          deviceId: input.deviceId,
          startedAt: input.startedAt,
          endedAt: input.endedAt,
          durationSeconds,
          capabilitiesUsed: {
             remoteInput: (input.peakPermissions?.remoteInput ?? 'disabled') === 'enabled',
             clipboardSync: input.peakPermissions?.clipboardSync ?? false,
             fileTransfer: input.peakPermissions?.fileTransfer ?? false,
          },
          unattended: input.unattended,
          counts: {
             inputBlocked: countEvents(input.events, 'input.blocked'),
             inputRateLimited: countEvents(input.events, 'input.rate_limited'),
             emergencyStops: countEvents(input.events, 'emergency_stop.engaged'),
             controlGrants: countEvents(input.events, 'control.granted'),
             controlRevokes: countEvents(input.events, 'control.revoked'),
          },
          endReason: input.endReason,
        };
   }



FILE: packages/shared/src/permissions/consentReceipt.ts
   import type { SessionSummary } from './sessionSummary';
   import type { SessionAuditEvent } from './sessionAuditEvents';

   /**
    * A verifiable consent receipt: the session summary + a hash CHAIN over the
       * session's audit timeline. Each line is hashed with the previous hash, so any
       * insertion/removal/reordering breaks the chain. The receipt is content-free and
       * verifiable OFFLINE (no server) by recomputing the chain from the line items.
       */
   export interface ReceiptLineItem {
     /** Sequence in the session timeline. */
        seq: number;
        /** Non-sensitive event type + ISO time (no payloads). */
        type: string;
        at: string;
        /** Hash linking this line to the previous (hex). */
        hash: string;
   }


   export interface ConsentReceipt {
     receiptId: string;
        summary: SessionSummary;
        lineItems: ReceiptLineItem[];
        /** Final chain hash = the merkle-ish root of the whole timeline. */
        rootHash: string;
        issuedAt: string;
        /** Schema/version tag so verifiers can evolve. */
        version: 1;
   }


   /** Canonical per-line preimage. Order-stable, non-sensitive fields only. */
  export function linePreimage(prevHash: string, seq: number, type: string, at: string): string {
       return `${prevHash}|${seq}|${type}|${at}`;
  }

  /**
      * Build the chained line items from the audit timeline using an injected hash fn
      * (so shared stays runtime-agnostic; desktop passes a SHA-256). Returns the items
      * + the root hash. Pure given the hash function.
      */
  export async function buildChain(
    events: SessionAuditEvent[],
       sha256Hex: (input: string) => Promise<string>,
       genesis = 'remotedesk-consent-genesis',
  ): Promise<{ lineItems: ReceiptLineItem[]; rootHash: string }> {
    let prev = genesis;
       const lineItems: ReceiptLineItem[] = [];
       for (let i = 0; i < events.length; i++) {
           const e = events[i];
           const hash = await sha256Hex(linePreimage(prev, i, e.type, e.at));
           lineItems.push({ seq: i, type: e.type, at: e.at, hash });
           prev = hash;
       }
       return { lineItems, rootHash: prev };
  }


  /**
   * Offline verification: recompute the chain from the receipt's own line items and
      * confirm every link + the root. Returns the first broken seq, or null if intact.
      * Requires NO server and NO original events — just the receipt.
   */
  export async function verifyReceipt(
       receipt: ConsentReceipt,
       sha256Hex: (input: string) => Promise<string>,
    genesis = 'remotedesk-consent-genesis',
  ): Promise<{ valid: boolean; firstBrokenSeq: number | null }> {
       let prev = genesis;
       for (const item of receipt.lineItems) {
           const expected = await sha256Hex(linePreimage(prev, item.seq, item.type, item.at));
           if (expected !== item.hash) return { valid: false, firstBrokenSeq: item.seq };
           prev = expected;
       }
       if (prev !== receipt.rootHash) return { valid: false, firstBrokenSeq: null };
       return { valid: true, firstBrokenSeq: null };
  }

   import {
     buildSessionSummary,
     type SessionAuditEvent,
     type SessionPermissionSet,
     type SessionSummary,
   } from '@remotedesk/shared';
   import type { SessionTimelineStore } from './sessionTimelineStore';


   /**
    * Builds a SessionSummary at disconnect from the Pack 18 timeline store + the
    * peak permission set observed during the session. Tracks the peak (most-enabled)
    * permissions so the summary reflects what was EVER active, not just the final state.
    */
   export class SessionSummaryBuilder {
     private startedAt = new Date().toISOString();
     private peak: Pick<SessionPermissionSet, 'remoteInput' | 'clipboardSync' | 'fileTransfer'> | null = null;
     private unattended = false;


     constructor(private readonly timeline: SessionTimelineStore, private readonly deviceId: string) {}

     markStarted(at: string = new Date().toISOString()): void {
       this.startedAt = at;
     }


     markUnattended(unattended: boolean): void {
       this.unattended = this.unattended || unattended;
     }


     /** Feed every permission snapshot; we keep the most-permissive seen. */
     observePermissions(p: SessionPermissionSet): void {
         const cur = this.peak;
         this.peak = {
           remoteInput: p.remoteInput === 'enabled' || cur?.remoteInput === 'enabled' ? 'enabled' : 'disabled',
           clipboardSync: !!p.clipboardSync || !!cur?.clipboardSync,
           fileTransfer: !!p.fileTransfer || !!cur?.fileTransfer,
         };
     }

     build(endReason: SessionSummary['endReason'], endedAt: string = new Date().toISOString()): SessionSummary {
       const events: SessionAuditEvent[] = [...this.timeline.all()];
         return buildSessionSummary({
           deviceId: this.deviceId,
           startedAt: this.startedAt,
           endedAt,
           events,
              peakPermissions: this.peak,
              unattended: this.unattended,
              endReason,
            });
        }
   }

   import {
        buildChain,
        verifyReceipt,
        type ConsentReceipt,
        type SessionAuditEvent,
     type SessionSummary,
   } from '@remotedesk/shared';


   /**
       * Issues + verifies consent receipts on the desktop using the renderer's SHA-256.
       * The receipt is user-held (downloadable JSON); a server copy is out of scope.
       * Verification works entirely offline from the receipt's own line items.
       */
   export class ConsentReceiptService {
     /** Build a receipt from a finished session's summary + audit timeline. */
        async issue(summary: SessionSummary, events: SessionAuditEvent[]): Promise<ConsentReceipt> {
          const { lineItems, rootHash } = await buildChain(events, sha256Hex);
            return {
              receiptId: cryptoRandomId(),
              summary,
              lineItems,
              rootHash,
              issuedAt: new Date().toISOString(),
              version: 1,
            };
        }


        /** Offline-verify a receipt (recompute the chain). */
        verify(receipt: ConsentReceipt): Promise<{ valid: boolean; firstBrokenSeq: number | null }> {
            return verifyReceipt(receipt, sha256Hex);
        }


        /** Trigger a browser download of the receipt JSON. */
        download(receipt: ConsentReceipt): void {
          const blob = new Blob([JSON.stringify(receipt, null, 2)], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `consent-receipt-${receipt.receiptId}.json`;
            a.click();
            URL.revokeObjectURL(url);
        }
   }
   async function sha256Hex(input: string): Promise<string> {
        const data = new TextEncoder().encode(input);
        const digest = await crypto.subtle.digest('SHA-256', data);
        return Array.from(new Uint8Array(digest), (b) => b.toString(16).padStart(2, '0')).join('');
   }


   function cryptoRandomId(): string {
        const bytes = new Uint8Array(16);
        crypto.getRandomValues(bytes);
        return Array.from(bytes, (b) => b.toString(16).padStart(2, '0')).join('');
   }

   import React, { useState } from 'react';
   import type { ConsentReceipt, SessionSummary } from '@remotedesk/shared';
   import type { ConsentReceiptService } from '../services/consentReceiptService';


   const REASON_LABEL: Record<string, string> = {
        user_disconnect: 'You ended the session',
        expired: 'Session time limit reached',
        reconnect_failed: 'Connection lost and could not safely resume',
        unknown: 'Session ended',
   };


   /**
    * Shown at disconnect. Displays the content-free summary and lets the user
       * download + verify a consent receipt. Screen view is already gone by this point.
       */
   export function SessionSummaryDialog({
     summary,
        receipt,
      service,
    onClose,
  }: {
      summary: SessionSummary;
      receipt: ConsentReceipt | null;
      service: ConsentReceiptService;
      onClose: () => void;
  }) {
    const [verifyState, setVerifyState] = useState<'idle' | 'valid' | 'invalid'>('idle');


      const mins = Math.floor(summary.durationSeconds / 60);
      const secs = summary.durationSeconds % 60;


      const verify = async () => {
        if (!receipt) return;
        const r = await service.verify(receipt);
        setVerifyState(r.valid ? 'valid' : 'invalid');
      };


      return (
        <div className="session-summary-dialog" role="dialog" aria-modal="true">
           <h2>Session ended</h2>
           <p className="session-summary-dialog__reason">{REASON_LABEL[summary.endReason]}</p>


           <dl className="session-summary-dialog__grid">
             <dt>Duration</dt><dd>{mins}m {secs}s</dd>
             <dt>Mode</dt><dd>{summary.unattended ? 'Unattended (pre-authorized)' : 'Attended'}</dd>
             <dt>Remote input</dt><dd>{summary.capabilitiesUsed.remoteInput ? 'used' : 'not used'}</dd>
             <dt>Clipboard</dt><dd>{summary.capabilitiesUsed.clipboardSync ? 'used' : 'not used'}</dd>
             <dt>File transfer</dt><dd>{summary.capabilitiesUsed.fileTransfer ? 'used' : 'not used'}</dd>
             <dt>Input blocked</dt><dd>{summary.counts.inputBlocked}</dd>
             <dt>Emergency stops</dt><dd>{summary.counts.emergencyStops}</dd>
           </dl>


           <div className="session-summary-dialog__actions">
             {receipt && (
               <>
                 <button type="button" className="btn btn--primary" onClick={() => service.download(receipt)}>
                   Download consent receipt
                 </button>
                 <button type="button" className="btn btn--ghost" onClick={verify}>
                {verifyState === 'idle' ? 'Verify receipt' : verifyState === 'valid' ? '✓ Verified intact' : '✗
  Verification failed'}
                 </button>
               </>
             )}
             <button type="button" className="btn" onClick={onClose}>Close</button>
          </div>
        </div>
      );
  }
