/**
    * Shared device access policy contract. The desktop enforces these; the API/web
    * already persist them. Everything here is plain data + string unions so it can
    * cross the data channel and be reasoned about in pure functions.
    */

export type DeviceTrustStatus = 'trusted' | 'untrusted' | 'blocked';

/** A point-in-time snapshot of a device's persisted access policy. */
export interface DeviceAccessPolicySnapshot {
  deviceId: string;
     trustStatus: DeviceTrustStatus;
     unattendedAccessEnabled: boolean;
     remoteInputEnabled: boolean;
     clipboardSyncEnabled: boolean;
     fileTransferEnabled: boolean;
     requiresSessionApproval: boolean;
     /** Hard cap in minutes; null = no cap. */
     maxSessionMinutes: number | null;
     /** ISO timestamp the policy was last updated server-side. */
     updatedAt: string;
}


/**
 * The effective, runtime permission set for a live session. Derived by the host
    * from the policy + host acceptance + emergency-stop state, then broadcast to the
    * viewer over the data channel. The HOST is the source of truth.
 */
export interface SessionPermissionSet {
     deviceId: string;
     remoteInput: RemoteInputPermissionState;
     clipboardSync: boolean;
     fileTransfer: boolean;
     /** ISO expiry derived from maxSessionMinutes, or null if uncapped. */
     sessionExpiresAt: string | null;
     /** True while the host emergency stop is engaged. */
     emergencyStopped: boolean;
     /** Monotonic version so the viewer ignores stale frames. */
     version: number;
     updatedAt: string;
}

/** Remote input is a tri-state so the UI can distinguish "off" from "not allowed". */
export type RemoteInputPermissionState = 'enabled' | 'disabled' | 'blocked';

/** Why a policy decision denied something. Drives user-facing messaging. */
export type PolicyDenialReason =
     | 'device_blocked'
     | 'device_untrusted'
     | 'missing_policy'
     | 'feature_disabled_by_policy'
     | 'host_not_accepted'
     | 'emergency_stop_active'
     | 'session_expired';


/** Result of a pure policy evaluation. */
   export interface PolicyDecision {
           allowed: boolean;
           /** Set only when allowed === false. */
           reason: PolicyDenialReason | null;
   }

   /** Convenience constructors keep call sites terse + consistent. */
   export function allow(): PolicyDecision {
     return { allowed: true, reason: null };
   }
   export function deny(reason: PolicyDenialReason): PolicyDecision {
           return { allowed: false, reason };
   }

   /**
    * Pure, fail-closed policy evaluation shared by the host (enforcement) and the
       * UI (gating). No I/O, no Date.now() side effects beyond explicit expiry math.
       *
       * Core rules:
       * - blocked device              => deny everything
       *    - untrusted device         => deny unattended access + remote input
       *    - missing policy (null)     => deny all risky features
       *    - remote input is OFF by default; host must accept first
       *    - host acceptance required for remote input / clipboard / file transfer
       */


   /** Gate session start itself. A blocked device cannot even start a session. */
   export function evaluateSessionStartPolicy(
     policy: DeviceAccessPolicySnapshot | null,
   ): PolicyDecision {
           if (!policy) return deny('missing_policy');
           if (policy.trustStatus === 'blocked') return deny('device_blocked');
           return allow();
   }

   /** Can remote input be enabled right now? */
   export function canEnableRemoteInput(
     policy: DeviceAccessPolicySnapshot | null,
           hostAccepted: boolean,
           emergencyStopped = false,
   ): PolicyDecision {
     if (!policy) return deny('missing_policy');
           if (policy.trustStatus === 'blocked') return deny('device_blocked');
           if (policy.trustStatus === 'untrusted') return deny('device_untrusted');
    if (!policy.remoteInputEnabled) return deny('feature_disabled_by_policy');
    if (emergencyStopped) return deny('emergency_stop_active');
    if (!hostAccepted) return deny('host_not_accepted');
    return allow();
}


/** Can clipboard sync be enabled? (Untrusted is allowed clipboard, not input.) */
export function canEnableClipboard(
  policy: DeviceAccessPolicySnapshot | null,
  hostAccepted: boolean,
): PolicyDecision {
    if (!policy) return deny('missing_policy');
    if (policy.trustStatus === 'blocked') return deny('device_blocked');
    if (!policy.clipboardSyncEnabled) return deny('feature_disabled_by_policy');
    if (!hostAccepted) return deny('host_not_accepted');
    return allow();
}


/** Can file transfer be enabled? */
export function canEnableFileTransfer(
  policy: DeviceAccessPolicySnapshot | null,
  hostAccepted: boolean,
): PolicyDecision {
    if (!policy) return deny('missing_policy');
    if (policy.trustStatus === 'blocked') return deny('device_blocked');
    if (!policy.fileTransferEnabled) return deny('feature_disabled_by_policy');
    if (!hostAccepted) return deny('host_not_accepted');
    return allow();
}


/** Human-readable message for a denial reason. UI-facing, non-sensitive. */
export function getPolicyDenialMessage(reason: PolicyDenialReason): string {
  switch (reason) {
        case 'device_blocked':
          return 'This device is blocked. No remote features are available.';
        case 'device_untrusted':
          return 'This device is untrusted. Remote input and unattended access are disabled.';
        case 'missing_policy':
          return 'Device policy could not be loaded. Risky features are disabled for safety.';
        case 'feature_disabled_by_policy':
          return 'This feature is disabled by the device access policy.';
        case 'host_not_accepted':
          return 'Waiting for the host to allow this feature.';
        case 'emergency_stop_active':
          return 'Emergency stop is active. Remote input is disabled.';
        case 'session_expired':
          return 'The session time limit has been reached.';
        default:
          return 'This action is not allowed by policy.';
    }
}

/** Derive a session expiry ISO timestamp from maxSessionMinutes. */
export function deriveSessionExpiry(
  policy: DeviceAccessPolicySnapshot | null,
  startedAt: Date = new Date(),
): string | null {
        if (!policy || policy.maxSessionMinutes === null) return null;
        return new Date(startedAt.getTime() + policy.maxSessionMinutes * 60_000).toISOString();
   }


   /** Is a session past its derived expiry? */
   export function isSessionExpired(
     sessionExpiresAt: string | null,
     now: Date = new Date(),
   ): boolean {
        if (!sessionExpiresAt) return false;
        return now.getTime() >= Date.parse(sessionExpiresAt);
   }


   /**
    * Build the effective permission set the host broadcasts to the viewer. This is
       * the single place runtime permissions are composed, so host + viewer never drift.
       */
   export function buildSessionPermissionSet(input: {
     policy: DeviceAccessPolicySnapshot | null;
        hostAccepted: boolean;
        emergencyStopped: boolean;
        startedAt?: Date;
        version: number;
   }): SessionPermissionSet {
     const { policy, hostAccepted, emergencyStopped, version } = input;
        const deviceId = policy?.deviceId ?? 'unknown';
        const expires = deriveSessionExpiry(policy, input.startedAt ?? new Date());


        const inputDecision = canEnableRemoteInput(policy, hostAccepted, emergencyStopped);
        let remoteInput: RemoteInputPermissionState;
        if (inputDecision.allowed) {
          remoteInput = 'enabled';
        } else if (
            inputDecision.reason === 'device_blocked' ||
            inputDecision.reason === 'device_untrusted'
        ) {
          remoteInput = 'blocked';
        } else {
          remoteInput = 'disabled';
        }


        return {
          deviceId,
            remoteInput,
            clipboardSync: canEnableClipboard(policy, hostAccepted).allowed,
            fileTransfer: canEnableFileTransfer(policy, hostAccepted).allowed,
            sessionExpiresAt: expires,
            emergencyStopped,
            version,
            updatedAt: new Date().toISOString(),
        };
   }

   /**
    * Fetches the persisted device access policy from the API. If the exact route
       * differs in your API, change ONLY this constant — everything else is decoupled.
       */
   export const DEVICE_POLICY_ENDPOINT = (deviceId: string): string =>
     `/api/devices/${encodeURIComponent(deviceId)}/policy`;


   export interface PolicyFetchResult {
        /** null when the policy could not be loaded (caller MUST fail closed). */
        policy: DeviceAccessPolicySnapshot | null;
        /** True if served from the display-only cache (not safe to enable features). */
        fromCache: boolean;
   }

   /**
    * Policy client. On any failure it returns `policy: null` so the evaluator fails
       * closed. The last known policy is cached ONLY for display (e.g. showing the last
       * trust status) and is never returned as the authoritative policy for enabling
       * risky features. No secrets are logged.
       */
   export class DevicePolicyClient {
     private displayCache: DeviceAccessPolicySnapshot | null = null;


        constructor(
          private readonly baseUrl: string,
          private readonly authToken: string,
        ) {}


        /** Authoritative fetch. Returns null policy on any error (fail closed). */
        async fetch(deviceId: string): Promise<PolicyFetchResult> {
          try {
            const res = await fetch(`${this.baseUrl}${DEVICE_POLICY_ENDPOINT(deviceId)}`, {
              headers: { authorization: `Bearer ${this.authToken}` },
            });
            if (!res.ok) throw new Error(`policy fetch status ${res.status}`);
            const policy = (await res.json()) as DeviceAccessPolicySnapshot;
            // Sanity: enforce the start gate so a blocked device is obvious to callers.
                evaluateSessionStartPolicy(policy);
                this.displayCache = policy;
                return { policy, fromCache: false };
            } catch (err) {
              // Non-sensitive structured diagnostic only.
                console.warn('[policy] fetch failed, failing closed', { deviceId, reason: errMessage(err) });
                return { policy: null, fromCache: false };
            }
        }


        /** Display-only last known policy. NEVER use this to enable risky features. */
        lastKnownForDisplay(): DeviceAccessPolicySnapshot | null {
          return this.displayCache;
        }
   }


   function errMessage(err: unknown): string {
        return err instanceof Error ? err.message : 'unknown';
   }

   import {
     buildSessionPermissionSet,
        type DeviceAccessPolicySnapshot,
        type SessionPermissionSet,
   } from '@remotedesk/shared';


   /**
    * Host-side authoritative permission state. Holds the policy + host-acceptance +
       * emergency-stop flags and produces a versioned SessionPermissionSet to broadcast.
       * The viewer only ever CONSUMES a permission set; it never computes its own.
    */
   export class HostPermissionState {
        private policy: DeviceAccessPolicySnapshot | null = null;
        private hostAccepted = false;
        private emergencyStopped = false;
        private version = 0;
        private startedAt = new Date();
        private listeners = new Set<(p: SessionPermissionSet) => void>();

        setPolicy(policy: DeviceAccessPolicySnapshot | null): void {
            this.policy = policy;
            this.emit();
        }


        /** Host explicitly accepts/relaxes risky features (e.g. clicks "Allow control"). */
        setHostAccepted(accepted: boolean): void {
            this.hostAccepted = accepted;
            this.emit();
        }


        setEmergencyStopped(stopped: boolean): void {
          this.emergencyStopped = stopped;
            this.emit();
        }
        markStarted(at: Date = new Date()): void {
          this.startedAt = at;
        }


        current(): SessionPermissionSet {
          return buildSessionPermissionSet({
             policy: this.policy,
             hostAccepted: this.hostAccepted,
             emergencyStopped: this.emergencyStopped,
             startedAt: this.startedAt,
              version: this.version,
            });
        }

        subscribe(fn: (p: SessionPermissionSet) => void): () => void {
          this.listeners.add(fn);
            return () => this.listeners.delete(fn);
        }


        private emit(): void {
            this.version += 1;
            const snapshot = this.current();
            for (const fn of this.listeners) fn(snapshot);
        }
   }

   import type { SessionPermissionSet } from '@remotedesk/shared';


   /**
       * Bidirectional remote-input gate.
       * - VIEWER side: only captures + sends input when permission is 'enabled'.
       * - HOST side: rejects any inbound input unless permission is 'enabled'.
       * Blocked input is logged as a structured, NON-SENSITIVE diagnostic (no key codes,
       * no coordinates beyond a count).
       */
   export class RemoteInputGate {
     private permissions: SessionPermissionSet | null = null;
        private blockedCount = 0;


        update(permissions: SessionPermissionSet | null): void {
          this.permissions = permissions;
        }


        /** True only when input is explicitly enabled and not emergency-stopped/expired. */
        inputAllowed(now: Date = new Date()): boolean {
            const p = this.permissions;
            if (!p) return false; // fail closed
            if (p.emergencyStopped) return false;
            if (p.remoteInput !== 'enabled') return false;
            if (p.sessionExpiresAt && now.getTime() >= Date.parse(p.sessionExpiresAt)) return false;
            return true;
        }


        /** VIEWER: call before sending each input event. Returns false to drop it. */
        guardOutbound(): boolean {
            if (this.inputAllowed()) return true;
            this.recordBlocked('outbound');
            return false;
        }
        /** HOST: call before executing each inbound input event. Returns false to reject. */
        guardInbound(): boolean {
          if (this.inputAllowed()) return true;
            this.recordBlocked('inbound');
            return false;
        }


        blockedTotal(): number {
          return this.blockedCount;
        }


        private recordBlocked(direction: 'inbound' | 'outbound'): void {
          this.blockedCount += 1;
            // Structured + non-sensitive: no event payload, just a counter + reason.
            if (this.blockedCount === 1 || this.blockedCount % 50 === 0) {
                console.info('[input-gate] blocked input', {
                  direction,
                  state: this.permissions?.remoteInput ?? 'none',
                  emergencyStopped: this.permissions?.emergencyStopped ?? false,
                  blockedTotal: this.blockedCount,
                });
            }
        }
   }

FILE: apps/desktop/src/renderer/src/services/emergencyStop.ts
   import type { HostPermissionState } from './permissionState';
   import { remoteInputGate, applyPermissionsToWebRtc } from './webrtc';
   import { sendPermissionSnapshot, type PermissionSnapshotFrame } from './sessionDataChannel';


   /**
       * Host emergency stop. Immediately disables remote input, rebroadcasts the
       * permission snapshot (so the viewer stops capturing), and updates local gate
       * state. Does NOT disconnect the WebRTC screen view.
       */
   export class EmergencyStop {
     private engaged = false;


        constructor(
            private readonly permissionState: HostPermissionState,
            private readonly send: (frame: PermissionSnapshotFrame) => void,
        ) {}


        isEngaged(): boolean {
          return this.engaged;
        }


        /** Engage: kill input now, broadcast, keep the screen view alive. */
        engage(): void {
            this.engaged = true;
            this.permissionState.setEmergencyStopped(true);
            const permissions = this.permissionState.current();
            applyPermissionsToWebRtc(permissions); // local gate denies immediately
            sendPermissionSnapshot(this.send, permissions); // tell the viewer to stop
        }

        /** Release: clears the stop. Input still requires host acceptance + policy. */
        release(): void {
          this.engaged = false;
            this.permissionState.setEmergencyStopped(false);
            const permissions = this.permissionState.current();
            applyPermissionsToWebRtc(permissions);
            sendPermissionSnapshot(this.send, permissions);
        }
   }

   import React from 'react';
   import {

  /**
      * Shows the current policy + permission status. Read-only; reflects the
      * authoritative permission set (host) or the applied snapshot (viewer).
   */
  export function PolicyStatusBanner({
       policy,
       permissions,
    fromCache,
  }: {
       policy: DeviceAccessPolicySnapshot | null;
       permissions: SessionPermissionSet | null;
    fromCache?: boolean;
  }) {
       const trust = policy?.trustStatus ?? 'unknown';
       const input = permissions?.remoteInput ?? 'disabled';


       const tone =
         trust === 'blocked' || input === 'blocked'
           ? 'danger'
            : input === 'enabled'
              ? 'ok'
              : 'warn';


       return (
         <div className={`policy-banner policy-banner--${tone}`} role="status">
            <span className="policy-banner__trust">Device: {trust}</span>
            <span className="policy-banner__sep"> · </span>
            <span className="policy-banner__input">Remote input: {input}</span>
            {permissions && (
              <>
                 <span className="policy-banner__sep"> · </span>
                 <span>Clipboard: {permissions.clipboardSync ? 'on' : 'off'}</span>
                 <span className="policy-banner__sep"> · </span>
                <span>File transfer: {permissions.fileTransfer ? 'on' : 'off'}</span>
              </>
            )}
            {permissions?.emergencyStopped && (
              <span className="policy-banner__estop"> · EMERGENCY STOP ACTIVE</span>
            )}
            {fromCache && <span className="policy-banner__cache"> · showing last known (offline)</span>}
            {policy === null && (
              <div className="policy-banner__reason">{getPolicyDenialMessage('missing_policy')}</div>
            )}
         </div>
       );
  }
