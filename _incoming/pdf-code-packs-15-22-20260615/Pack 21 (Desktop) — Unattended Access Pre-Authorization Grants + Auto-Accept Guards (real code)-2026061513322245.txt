   import { deny, allow, type DeviceAccessPolicySnapshot, type PolicyDecision } from './policyEvaluator';
   import type { FeatureKey } from './devicePolicy';

   /**
    * A time-boxed, scope-limited pre-authorization that lets a trusted device start
       * an UNATTENDED session (no human at the host to consent). Grants are revocable
       * and expire. They never widen what the policy allows; they only substitute for
       * live host consent, and only for the features explicitly listed.
       */
   export interface UnattendedGrant {
     grantId: string;
           deviceId: string;
           /** Which interactive features this grant pre-authorizes (subset of policy). */
           scope: FeatureKey[];
           /** ISO start; grant is not valid before this. */
           notBefore: string;
           /** ISO expiry; grant is invalid at/after this. */
           expiresAt: string;
           /** Set when an operator revokes early. */
           revokedAt: string | null;
           /** Non-sensitive label of who created it (display/audit only). */
           createdByLabel: string | null;
   }

   export type UnattendedDenialReason =
           | 'no_grant'
           | 'grant_revoked'
           | 'grant_not_yet_valid'
           | 'grant_expired'
           | 'device_not_trusted'
           | 'unattended_disabled_by_policy';


   export interface UnattendedDecision {
           allowed: boolean;
           reason: UnattendedDenialReason | null;
           /** The effective scope the grant authorizes (empty when denied). */
           scope: FeatureKey[];
   }


   /** Is the grant valid at `now` (time window + not revoked)? Pure. */
   export function isGrantActive(grant: UnattendedGrant, now: Date = new Date()): boolean {
           if (grant.revokedAt) return false;
           const t = now.getTime();
           return t >= Date.parse(grant.notBefore) && t < Date.parse(grant.expiresAt);
   }


   /**
       * Evaluate whether unattended access may proceed. Fail-closed and never broader
       * than the policy:
       *    - policy must enable unattended access AND device must be trusted
       *    - a non-revoked, in-window grant must exist
       * - effective scope = grant.scope intersected with policy-permitted features
       */
export function evaluateUnattendedAccess(input: {
    policy: DeviceAccessPolicySnapshot | null;
    grant: UnattendedGrant | null;
  now?: Date;
}): UnattendedDecision {
    const { policy, grant } = input;
    const now = input.now ?? new Date();


    if (!policy || policy.trustStatus === 'blocked' || policy.trustStatus === 'untrusted') {
        return { allowed: false, reason: 'device_not_trusted', scope: [] };
    }
    if (!policy.unattendedAccessEnabled) {
      return { allowed: false, reason: 'unattended_disabled_by_policy', scope: [] };
    }
    if (!grant) return { allowed: false, reason: 'no_grant', scope: [] };
    if (grant.revokedAt) return { allowed: false, reason: 'grant_revoked', scope: [] };
    if (now.getTime() < Date.parse(grant.notBefore)) {
        return { allowed: false, reason: 'grant_not_yet_valid', scope: [] };
    }
    if (now.getTime() >= Date.parse(grant.expiresAt)) {
      return { allowed: false, reason: 'grant_expired', scope: [] };
    }

    // Intersect grant scope with what the policy actually permits (never broader).
    const policyAllowed = policyPermittedFeatures(policy);
    const scope = grant.scope.filter((f) => policyAllowed.includes(f));
    return { allowed: true, reason: null, scope };
}


/** Features the policy itself permits (used to clamp grant scope). */
export function policyPermittedFeatures(policy: DeviceAccessPolicySnapshot): FeatureKey[] {
    const out: FeatureKey[] = [];
    if (policy.remoteInputEnabled) out.push('remoteInput' as FeatureKey);
    if (policy.clipboardSyncEnabled) out.push('clipboardSync' as FeatureKey);
    if (policy.fileTransferEnabled) out.push('fileTransfer' as FeatureKey);
    if (policy.unattendedAccessEnabled) out.push('unattendedAccess' as FeatureKey);
    return out;
}


export function unattendedDenialMessage(reason: UnattendedDenialReason): string {
  switch (reason) {
        case 'no_grant': return 'No unattended-access grant exists for this device.';
        case 'grant_revoked': return 'The unattended-access grant was revoked.';
        case 'grant_not_yet_valid': return 'The unattended-access grant is not active yet.';
        case 'grant_expired': return 'The unattended-access grant has expired.';
        case 'device_not_trusted': return 'Unattended access requires a trusted device.';
        case 'unattended_disabled_by_policy': return 'Unattended access is disabled by policy.';
        default: return 'Unattended access is not allowed.';
    }
}


/** Bridge to a PolicyDecision-style result for callers that want it. */
export function unattendedToPolicyDecision(d: UnattendedDecision): PolicyDecision {
    return d.allowed ? allow() : deny('host_not_accepted');
}

   import { evaluateUnattendedAccess, type UnattendedGrant } from './unattendedGrant';
   import type { DeviceAccessPolicySnapshot } from './devicePolicy';
   import type { SessionApprovalSnapshot } from './sessionApproval';

   /**
       * Decides whether a session may auto-accept (skip the live consent prompt) under
       * an unattended grant. This is the ONLY sanctioned way to satisfy consent without
       * a human. Fail-closed: anything short of a valid grant on a trusted device with
       * unattended enabled falls back to normal (pending) consent.
    */
   export interface AutoAcceptResult {
        /** True only when a valid unattended grant authorizes auto-accept. */
        autoAccepted: boolean;
        /** The approval snapshot to apply (approved when auto-accepted, else pending). */
        approval: SessionApprovalSnapshot;
   }


   export function evaluateAutoAccept(input: {
     policy: DeviceAccessPolicySnapshot | null;
        grant: UnattendedGrant | null;
        requestedByLabel?: string | null;
     now?: Date;
   }): AutoAcceptResult {
        const decision = evaluateUnattendedAccess({ policy: input.policy, grant: input.grant, now: input.now });
        const at = (input.now ?? new Date()).toISOString();


        if (decision.allowed) {
          return {
            autoAccepted: true,
             approval: { status: 'approved', decidedAt: at, requestedByLabel: input.requestedByLabel ?? 'unattended grant' },
          };
        }
        // Fall back to normal consent: pending, requires a human.
        return {
          autoAccepted: false,
          approval: { status: 'pending', decidedAt: null, requestedByLabel: input.requestedByLabel ?? null },
        };
   }

   import {
        isGrantActive,
        evaluateUnattendedAccess,
        type DeviceAccessPolicySnapshot,
        type UnattendedDecision,
     type UnattendedGrant,
   } from '@remotedesk/shared';

   /**
       * Loads + caches the active unattended grant for a device and evaluates whether
       * unattended access may proceed. Fail-closed: a fetch failure yields no grant
       * (=> denied), and the store never fabricates a grant. Grants can be revoked
       * locally (immediately) in addition to server-side revocation.
    */
   export interface GrantSource {
        /** Fetch the current grant for a device, or null. Throwing => treated as null. */
        fetchGrant: (deviceId: string) => Promise<UnattendedGrant | null>;
        /** Persist a revocation server-side. */
        revokeGrant: (grantId: string) => Promise<void>;
   }


   export class UnattendedGrantStore {
     private grant: UnattendedGrant | null = null;
        private listeners = new Set<(g: UnattendedGrant | null) => void>();

        constructor(private readonly source: GrantSource) {}


        /** Load the grant for a device. On any error => null (fail closed). */
        async load(deviceId: string): Promise<UnattendedGrant | null> {
            try {
              this.grant = await this.source.fetchGrant(deviceId);
            } catch {
              this.grant = null;
            }
            this.emit();
            return this.grant;
        }


        current(): UnattendedGrant | null {
            return this.grant;
        }


        /** Is there an active (in-window, non-revoked) grant right now? */
        hasActiveGrant(now: Date = new Date()): boolean {
          return !!this.grant && isGrantActive(this.grant, now);
        }


        /** Evaluate unattended access against a policy + the cached grant. */
        evaluate(policy: DeviceAccessPolicySnapshot | null, now: Date = new Date()): UnattendedDecision {
            return evaluateUnattendedAccess({ policy, grant: this.grant, now });
        }
        /** Revoke immediately: clears locally + persists. Local clear is fail-safe. */
        async revoke(): Promise<void> {
          const g = this.grant;
            this.grant = g ? { ...g, revokedAt: new Date().toISOString() } : null;
            this.emit();
            if (g) {
              try {
                  await this.source.revokeGrant(g.grantId);
                } catch {
                    // Local revocation already applied; server retry is the caller's concern.
                }
            }
        }


        subscribe(fn: (g: UnattendedGrant | null) => void): () => void {
            this.listeners.add(fn);
            return () => this.listeners.delete(fn);
        }


        private emit(): void {
          for (const fn of this.listeners) fn(this.grant);
        }
   }

   import {
     evaluateAutoAccept,
        makeAuditEvent,
        type DeviceAccessPolicySnapshot,
        type SessionApprovalSnapshot,
        type SessionAuditEvent,
     type UnattendedGrant,
   } from '@remotedesk/shared';


   /**
       * Bridges an unattended grant to the Pack 16 session-approval flow. At session
       * start it decides whether to auto-accept (under a valid grant) or fall back to a
       * pending human prompt. It emits a non-sensitive audit event either way and
       * re-evaluates on demand (e.g. after a reconnect re-check or grant revocation).
    */
   export interface AutoAcceptDeps {
        /** Apply the resulting approval snapshot to the host approval state. */
        applyApproval: (approval: SessionApprovalSnapshot) => void;
        onAudit?: (e: SessionAuditEvent) => void;
   }


   export class AutoAcceptController {
        constructor(private readonly deps: AutoAcceptDeps) {}


        /** Returns true if the session auto-accepted under a grant. */
        evaluate(input: {
            policy: DeviceAccessPolicySnapshot | null;
            grant: UnattendedGrant | null;
            requestedByLabel?: string | null;
            now?: Date;
        }): boolean {
          const result = evaluateAutoAccept(input);
            this.deps.applyApproval(result.approval);
            this.deps.onAudit?.(
              makeAuditEvent(result.autoAccepted ? 'control.granted' : 'control.revoked', {
                mode: result.autoAccepted ? 'unattended_auto_accept' : 'attended_pending',
              }),
            );
            return result.autoAccepted;
        }
   }

   import React, { useEffect, useState } from 'react';
   import { isGrantActive, type UnattendedGrant } from '@remotedesk/shared';
   import type { UnattendedGrantStore } from '../services/unattendedGrantStore';


   /**
    * Shows the current unattended-access grant + a revoke control. Read-only except
       * for Revoke, which kills auto-accept immediately (local + server).
       */
   export function UnattendedGrantPanel({ store }: { store: UnattendedGrantStore }) {
     const [grant, setGrant] = useState<UnattendedGrant | null>(store.current());


        useEffect(() => store.subscribe(setGrant), [store]);


        if (!grant) {
       return <div className="unattended-panel unattended-panel--none">No unattended-access grant. Sessions require live
   approval.</div>;
        }
        const active = isGrantActive(grant);
       return (
         <div className={`unattended-panel ${active ? 'unattended-panel--active' : 'unattended-panel--inactive'}`}>
           <div className="unattended-panel__row">
               <span>Unattended grant: {active ? 'active' : grant.revokedAt ? 'revoked' : 'inactive'}</span>
               <span className="unattended-panel__exp">expires {new Date(grant.expiresAt).toLocaleString()}</span>
            </div>
            <div className="unattended-panel__scope">scope: {grant.scope.join(', ') || 'none'}</div>
            {active && (
              <button type="button" className="btn btn--danger" onClick={() => void store.revoke()}>
                 Revoke unattended access
               </button>
           )}
         </div>
       );
  }

  import React from 'react';
  import { unattendedDenialMessage, type UnattendedDecision } from '@remotedesk/shared';


  /**
      * Banner shown while a session is running unattended (auto-accepted) OR explaining
      * why unattended access was denied and the session fell back to live approval.
   */
  export function UnattendedBanner({ decision }: { decision: UnattendedDecision }) {
       if (decision.allowed) {
         return (
            <div className="unattended-banner unattended-banner--active" role="status">
              Unattended session active · scope: {decision.scope.join(', ') || 'view only'}
            </div>
         );
       }
       if (!decision.reason) return null;
       return (
         <div className="unattended-banner unattended-banner--denied" role="status">
           {unattendedDenialMessage(decision.reason)} Falling back to live approval.
         </div>
       );
  }
