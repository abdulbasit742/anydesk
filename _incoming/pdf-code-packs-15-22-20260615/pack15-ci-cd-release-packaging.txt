  import type { ChannelName } from './types';


  /** Canonical channel order: lower = earlier/riskier. */
  export const CHANNEL_ORDER: Readonly<Record<ChannelName, number>> = {
       canary: 1,
       beta: 2,
       stable: 3,
  };


  export const ALL_CHANNELS: readonly ChannelName[] = ['canary', 'beta', 'stable'];


  export function isChannel(value: string): value is ChannelName {
       return (ALL_CHANNELS as readonly string[]).includes(value);
  }


  /** Stable requires the most scrutiny; used by readiness to pick gate strictness. */
  export function isStrictChannel(channel: ChannelName): boolean {
    return channel === 'stable';
  }


  /** Can a build on `from` be promoted to `to`? Only one step up at a time. */
  export function canPromote(from: ChannelName, to: ChannelName): boolean {
       return CHANNEL_ORDER[to] - CHANNEL_ORDER[from] === 1;
  }


packages/shared/src/release/buildArtifact.ts       — SAFE_DIRECT_COPY
  import type { BuildArtifactDto, SigningStatus } from './types';


  /** A sha256 hex string is 64 lowercase hex chars. */
  export const SHA256_RE = /^[a-f0-9]{64}$/;


  export function isValidSha256(hash: string): boolean {
       return SHA256_RE.test(hash);
  }


  /** Basic semver-ish check (major.minor.patch with optional prerelease). */
  export const SEMVER_RE = /^\d+\.\d+\.\d+(?:-[0-9A-Za-z.-]+)?$/;


  export function isValidVersion(version: string): boolean {
    return SEMVER_RE.test(version);
  }


  /** Is this artifact considered signed? NEVER true unless explicitly 'signed'. */
  export function isSigned(signing: SigningStatus): boolean {
       return signing === 'signed';
  }


  /** A release is fully signed only if every artifact is signed (and there is ≥1). */
  export function isFullySigned(artifacts: Pick<BuildArtifactDto, 'signing'>[]): boolean {
    return artifacts.length > 0 && artifacts.every((a) => isSigned(a.signing));
  }


packages/shared/src/release/smokeResult.ts       — SAFE_DIRECT_COPY
  import type { SmokeCheck, SmokeOutcome } from './types';


  /**
      * Pure smoke-result aggregation. The outcome is derived from the checks: a run
      * passes ONLY if there is at least one check and every check passed. An empty
      * check list is treated as a FAIL (no fake success).
      */
  export function deriveOutcome(checks: SmokeCheck[]): SmokeOutcome {
    if (checks.length === 0) return 'failed';
       return checks.every((c) => c.passed) ? 'passed' : 'failed';
  }


  export function failedChecks(checks: SmokeCheck[]): SmokeCheck[] {
       return checks.filter((c) => !c.passed);
  }
  /** Summarize for display: "3/4 passed". */
  export function summarizeChecks(checks: SmokeCheck[]): string {
          const passed = checks.filter((c) => c.passed).length;
          return `${passed}/${checks.length} passed`;
  }


packages/shared/src/release/readiness.ts           — SAFE_DIRECT_COPY
  import type {
          BuildArtifactDto,
          ChannelName,
          ReadinessReport,
          SmokeTestRunDto,
  } from './types';
  import { isFullySigned } from './buildArtifact';

  /**
      * Pure readiness evaluation shared by API + tests. Refuses to report ready unless
      * ALL gates pass. No fake success: missing data is a blocker, not a pass.
      *
      * Gates:
      *       - the channel must exist + be enabled
      *       - there must be a signed desktop build (all artifacts signed) for the version
      * - the latest smoke run for the version must have passed
      */
  export function evaluateReadiness(input: {
    channel: ChannelName;
          channelExists: boolean;
          channelEnabled: boolean;
          version: string | null;
          artifacts: Pick<BuildArtifactDto, 'signing'>[];
    latestSmokeRun: Pick<SmokeTestRunDto, 'outcome'> | null;
  }): ReadinessReport {
          const blockers: string[] = [];


          if (!input.channelExists) blockers.push(`channel "${input.channel}" does not exist`);
          else if (!input.channelEnabled) blockers.push(`channel "${input.channel}" is disabled`);


          if (!input.version) {
              blockers.push('no release version selected');
          }


          if (input.artifacts.length === 0) {
            blockers.push('no build artifacts for this version');
          } else if (!isFullySigned(input.artifacts)) {
              blockers.push('desktop build is not fully signed');
          }


          if (!input.latestSmokeRun) {
            blockers.push('no smoke run recorded for this version');
          } else if (input.latestSmokeRun.outcome !== 'passed') {
              blockers.push('latest smoke run did not pass');
          }


          return {
              ready: blockers.length === 0,
        channel: input.channel,
        version: input.version,
        blockers,
        checkedAt: new Date().toISOString(),
      };
  }

scripts/release/check-release-readiness.mjs        — SAFE_DIRECT_COPY
  #!/usr/bin/env node
  /**
      * Advisory release-readiness probe for CI. Hits the readiness endpoint if
      * RELEASE_API_URL + RELEASE_API_TOKEN are set; otherwise prints a notice and
      * exits 0 (never blocks a normal PR). Never fakes a green result.
      */
  const advisory = process.argv.includes('--advisory');
  const base = process.env.RELEASE_API_URL;
  const token = process.env.RELEASE_API_TOKEN;
  const channel = process.env.RELEASE_CHANNEL ?? 'stable';


  async function main() {
       if (!base || !token) {
         console.log('[readiness] RELEASE_API_URL/TOKEN not set — skipping (advisory).');
           process.exit(0);
       }
       const res = await fetch(`${base}/api/releases/readiness?channel=${channel}`, {
         headers: { authorization: `Bearer ${token}` },
       });
       if (!res.ok) {
           console.log(`[readiness] endpoint returned ${res.status}`);
           process.exit(advisory ? 0 : 1);
       }

scripts/release/verify-desktop-artifact.mjs        — SAFE_DIRECT_COPY
  #!/usr/bin/env node
  /**
   * Verifies a built desktop artifact directory: checks files exist and computes
      * sha256 for each. Does NOT verify signatures (signing is REVIEW_REQUIRED). Prints
      * a manifest you can POST to /api/releases/artifacts. Never claims 'signed'.
   */
  import { readdir, stat, readFile } from 'node:fs/promises';
  import { createHash } from 'node:crypto';
  import { join } from 'node:path';


  const dirArg = process.argv.indexOf('--dir');
  const dir = dirArg !== -1 ? process.argv[dirArg + 1] : 'apps/desktop/dist';


  async function sha256(path) {
    const buf = await readFile(path);
       return createHash('sha256').update(buf).digest('hex');
  }


  async function main() {
    let entries;
       try {
         entries = await readdir(dir);
       } catch {
         console.log(`[verify] no artifact dir at ${dir} — skipping.`);
           process.exit(0);
       }
       const manifest = [];
       for (const name of entries) {
           const full = join(dir, name);
           const s = await stat(full);
           if (!s.isFile()) continue;
           if (!/\.(dmg|exe|AppImage|zip|deb|blockmap)$/i.test(name)) continue;
           manifest.push({ file: name, sizeBytes: s.size, sha256: await sha256(full), signing: 'unsigned' });
       }
       console.log(JSON.stringify({ dir, artifacts: manifest }, null, 2));
       process.exit(0);
  }


  main().catch((err) => { console.error('[verify] error:', err.message); process.exit(0); });

  #!/usr/bin/env node
  /**
   * Minimal local smoke check. Runs a few cheap, real assertions about the repo
      * build outputs and prints a checks array compatible with POST /releases/smoke-runs.
      * Advisory by default — never fakes a pass. Each check reflects a real condition.
   */
  import { access } from 'node:fs/promises';
  import { constants } from 'node:fs';


  const advisory = process.argv.includes('--advisory');


  async function exists(path) {
    try { await access(path, constants.F_OK); return true; } catch { return false; }
  }


  async function main() {
    const checks = [
           { name: 'shared package built', passed: await exists('packages/shared/dist') },
           { name: 'api build present', passed: await exists('apps/api/dist') || await exists('apps/api/build') },
           { name: 'web build present', passed: await exists('apps/web/.next') },
       ];
       const allPassed = checks.length > 0 && checks.every((c) => c.passed);
       console.log(JSON.stringify({ outcome: allPassed ? 'passed' : 'failed', checks }, null, 2));
       // Advisory mode never blocks; strict mode exits non-zero on failure.
       process.exit(allPassed || advisory ? 0 : 1);
  }


  main().catch((err) => { console.error('[smoke] error:', err.message); process.exit(advisory ? 0 : 1); });

  # Release Packaging & CI


  Release-channel, build-artifact, and smoke-QA foundations plus CI that runs
  keyless on normal PRs. **This is NOT production-ready release tooling:** signing,
  notarization, and deployment are not wired (they need certificates/secrets) and
  nothing here fakes a signed build or a successful deploy.


  ## Concepts
  - **ReleaseChannel** — canary / beta / stable, ordered; promotion is one step at a time.
  - **BuildArtifact** — version + platform + sha256 + size + signing status. Signing
    defaults to `unsigned` and is only `signed` when the pipeline explicitly says so.
  - **DesktopRelease** — a versioned set of artifacts on a channel; `fullySigned` is
    DERIVED (`isFullySigned`) from the real artifact signing states.
  - **SmokeTestRun** — named checks; outcome is DERIVED (`deriveOutcome`) and an empty
    list fails. No fake success.


  ## Readiness
  `GET /api/releases/readiness?channel=stable[&version=x]` refuses to report ready
  unless: the channel exists + is enabled, a signed desktop build exists for the
  version, AND the latest smoke run passed. Missing data is a blocker, not a pass.
  Every check is audit-logged.


  ## CI
  - `api-web-ci.yml` — keyless: install, typecheck/build/test where scripts exist,
    advisory readiness probe that never fails the PR.
  - `desktop-build-smoke.yml` — builds an UNSIGNED example artifact + runs a local
    smoke check. **Signing/notarization steps are commented and REVIEW_REQUIRED**
    because they need certificates.


  ## Scripts
  - `check-release-readiness.mjs` — advisory probe of the readiness endpoint.
  - `verify-desktop-artifact.mjs` — hashes built artifacts; never claims signed.
  - `run-local-smoke-check.mjs` — cheap real build-output checks; advisory.


docs/desktop-signing-readiness.md     — SAFE_DIRECT_COPY
  # Desktop Signing Readiness


  What exists, and exactly what is still required before a build can be called
  "signed" and shipped. **No certificates are included. No build is faked as signed.**


  ## Current state
  - The desktop reports `signing: 'signed'` ONLY when the build pipeline sets
    `REMOTEDESK_BUILD_SIGNED=1` (see `main/release/buildInfo.ts`). Default is `unsigned`.
  - `BuildArtifact.signing` defaults to `unsigned`; the API never fabricates `signed`.
  - `DesktopRelease.fullySigned` is derived from artifact states, not asserted.
  - Readiness blocks on any unsigned artifact.


  ## Required for real signing (REVIEW_REQUIRED — needs secrets)
  1. **macOS**: Developer ID Application cert (.p12) + notarization (Apple ID / API key).
     Wire into `desktop-build-smoke.yml` (commented section), set `REMOTEDESK_BUILD_SIGNED=1`
     only after the sign + notarize + staple steps succeed.
  2. **Windows**: code-signing cert (.pfx) or cloud HSM; sign the .exe + installer.
  3. **Linux**: optional GPG signing of the AppImage/deb.
  4. After signing, POST each artifact to `/api/releases/artifacts` with `signing: 'signed'`
     and its real sha256 (use `verify-desktop-artifact.mjs` to compute hashes).


  ## Do NOT
  - Do not set `REMOTEDESK_BUILD_SIGNED=1` unless signing actually ran.
  - Do not POST `signing: 'signed'` for an unsigned artifact.
  - Do not claim production readiness until signing + smoke + deploy paths are wired.
