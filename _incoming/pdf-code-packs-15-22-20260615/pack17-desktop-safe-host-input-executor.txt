   /**
       * Normalized inbound input action. The HOST executor reasons about these, not raw
       * DOM/native events, so gating + rate limiting + auditing are payload-agnostic and
       * never need to inspect (or log) actual key codes or coordinates.
       */
   export type InputActionKind =
     | 'mouse_move'
        | 'mouse_button'
        | 'mouse_scroll'
        | 'key'
        | 'text';


   export interface InputAction {
        kind: InputActionKind;
        /** Opaque, already-validated payload. Treated as a black box by gates/audit. */
        payload: unknown;
        /** Monotonic client sequence for ordering/dedupe (no timing secrets). */
        seq: number;
   }

   /** Which kinds are "high-frequency" and should be rate-limited more loosely. */
   export const HIGH_FREQUENCY_KINDS: readonly InputActionKind[] = ['mouse_move', 'mouse_scroll'];

   export function isHighFrequency(kind: InputActionKind): boolean {
     return HIGH_FREQUENCY_KINDS.includes(kind);
   }

   /** A non-sensitive descriptor for logs/audit: kind + seq only, never the payload. */
   export function describeAction(action: InputAction): { kind: InputActionKind; seq: number } {
        return { kind: action.kind, seq: action.seq };
   }

   /**
    * Session-local audit events for control/permission changes. These are
       * NON-SENSITIVE by construction: no payloads, no keystrokes, no tokens. They can
       * be forwarded to the API audit log (Pack 14) by the host if desired.
    */
   export type SessionAuditType =
        | 'control.granted'
        | 'control.revoked'
        | 'emergency_stop.engaged'
        | 'emergency_stop.released'
        | 'input.blocked'
        | 'input.rate_limited'
        | 'session.expired';

   export interface SessionAuditEvent {
     type: SessionAuditType;
        /** ISO timestamp. */
        at: string;
        /** Small, non-sensitive detail (counts, reasons, kinds). NEVER payloads. */
        detail: Record<string, string | number | boolean>;
   }

   export function makeAuditEvent(
     type: SessionAuditType,
        detail: Record<string, string | number | boolean> = {},
        now: Date = new Date(),
   ): SessionAuditEvent {
     return { type, at: now.toISOString(), detail };
   }


   /** Guard: reject any detail key that looks sensitive (defense in depth). */
   const FORBIDDEN = [/token/i, /password/i, /secret/i, /authorization/i, /cookie/i, /key$/i, /payload/i];

   export function isAuditDetailSafe(detail: Record<string, unknown>): boolean {
        return !Object.keys(detail).some((k) => FORBIDDEN.some((re) => re.test(k)));
   }

FILE: apps/desktop/src/renderer/src/services/inputRateLimiter.ts
   import { isHighFrequency, type InputActionKind } from '@remotedesk/shared';

   /**
       * Token-bucket rate limiter for inbound input actions. High-frequency kinds
       * (mouse move/scroll) get a larger bucket; discrete kinds (key/text/button) get a
       * tighter one to blunt scripted floods. Pure + deterministic given a clock.
       */
   export interface RateLimiterOptions {
     /** Tokens per second for discrete actions (key/button/text). */
        discreteRatePerSec?: number;
        /** Tokens per second for high-frequency actions (move/scroll). */
        highFreqRatePerSec?: number;
        /** Max burst (bucket capacity) multiplier applied to the per-sec rate. */
        burstFactor?: number;
   }


   interface Bucket {
        tokens: number;
        capacity: number;
        ratePerMs: number;
        lastRefill: number;
   }


   export class InputRateLimiter {
     private discrete: Bucket;
        private highFreq: Bucket;


        constructor(opts: RateLimiterOptions = {}, now: number = Date.now()) {
            const discreteRate = opts.discreteRatePerSec ?? 25;
            const highFreqRate = opts.highFreqRatePerSec ?? 200;
            const burst = opts.burstFactor ?? 2;
            this.discrete = this.makeBucket(discreteRate, burst, now);
            this.highFreq = this.makeBucket(highFreqRate, burst, now);
        }


        /** Returns true if the action is allowed; false if it should be dropped. */
        allow(kind: InputActionKind, now: number = Date.now()): boolean {
          const bucket = isHighFrequency(kind) ? this.highFreq : this.discrete;
            this.refill(bucket, now);
            if (bucket.tokens >= 1) {
             bucket.tokens -= 1;
             return true;
            }
            return false;
        }

        private makeBucket(ratePerSec: number, burstFactor: number, now: number): Bucket {
          const capacity = ratePerSec * burstFactor;
            return { tokens: capacity, capacity, ratePerMs: ratePerSec / 1000, lastRefill: now };
        }


        private refill(bucket: Bucket, now: number): void {
            const elapsed = now - bucket.lastRefill;
            if (elapsed <= 0) return;
            bucket.tokens = Math.min(bucket.capacity, bucket.tokens + elapsed * bucket.ratePerMs);
            bucket.lastRefill = now;
        }
   }

   import {
     describeAction,
        makeAuditEvent,
        type InputAction,
     type SessionAuditEvent,
   } from '@remotedesk/shared';
   import type { RemoteInputGate } from './remoteInputGate';
   import { InputRateLimiter } from './inputRateLimiter';


   /**
       * Optional native sink. By DEFAULT there is none, so execution is a safe no-op.
       * A real OS executor may be injected ONLY if the project already ships one; this
       * shell never imports or enables native input itself.
       */
   export interface NativeInputSink {
     execute(action: InputAction): void;
   }
export interface HostInputExecutorOptions {
  /** Inject a real native sink ONLY if a safe one exists. Default: none (no-op). */
        nativeSink?: NativeInputSink;
        rateLimiter?: InputRateLimiter;
        /** Emit audit events (engage/block/rate-limit). */
        onAudit?: (event: SessionAuditEvent) => void;
}

/**
 * The single host-side choke point for inbound input. Every inbound action MUST
    * go through execute(). Order of checks (all fail-closed):
    *   1. policy/permission gate (Pack 15 RemoteInputGate)
    *       2. rate limit
    *       3. native sink (no-op unless explicitly injected)
    * Blocked + rate-limited actions are counted and audited (kind + seq only).
    */
export class HostInputExecutor {
  private readonly rateLimiter: InputRateLimiter;
        private executed = 0;
        private blocked = 0;
        private rateLimited = 0;


        constructor(
          private readonly gate: RemoteInputGate,
          private readonly opts: HostInputExecutorOptions = {},
        ) {
            this.rateLimiter = opts.rateLimiter ?? new InputRateLimiter();
        }

        /** Process one inbound action. Returns the disposition for callers/tests. */
        execute(action: InputAction, now: number = Date.now()): 'executed' | 'blocked' | 'rate_limited' {
          // 1. Policy/permission gate — fail closed.
            if (!this.gate.guardInbound()) {
              this.blocked += 1;
             this.emitBlocked('policy', action);
             return 'blocked';
            }
            // 2. Rate limit.
            if (!this.rateLimiter.allow(action.kind, now)) {
              this.rateLimited += 1;
             this.emitRateLimited(action);
             return 'rate_limited';
            }
            // 3. Native sink — NO-OP unless a safe executor was explicitly injected.
            this.opts.nativeSink?.execute(action);
            this.executed += 1;
            return 'executed';
        }


        stats(): { executed: number; blocked: number; rateLimited: number } {
            return { executed: this.executed, blocked: this.blocked, rateLimited: this.rateLimited };
        }


        private emitBlocked(reason: string, action: InputAction): void {
            if (this.blocked === 1 || this.blocked % 50 === 0) {
              this.opts.onAudit?.(makeAuditEvent('input.blocked', { reason, ...describeAction(action), blockedTotal: this.blocked
   }));
            }
        }


        private emitRateLimited(action: InputAction): void {
            if (this.rateLimited === 1 || this.rateLimited % 100 === 0) {
              this.opts.onAudit?.(makeAuditEvent('input.rate_limited', { ...describeAction(action), rateLimitedTotal:
   this.rateLimited }));
       }
        }
   }

FILE: apps/desktop/src/renderer/src/services/sessionAuditHooks.ts
   import {
     isAuditDetailSafe,
     type SessionAuditEvent,
   } from '@remotedesk/shared';


   /**
       * Collects session audit events locally and optionally forwards them to a sink
       * (e.g. the Pack 14 AuditClient). Drops any event whose detail looks sensitive
       * (defense in depth) and never logs payloads. Forwarding failures are swallowed
       * so auditing never breaks a session.
    */
   export type AuditSink = (event: SessionAuditEvent) => void | Promise<void>;


   export class SessionAuditHooks {
        private buffer: SessionAuditEvent[] = [];


        constructor(private readonly sink?: AuditSink, private readonly maxBuffer = 500) {}


        /** Record an event. Sensitive-looking detail is rejected outright. */
        record(event: SessionAuditEvent): void {
            if (!isAuditDetailSafe(event.detail)) {
              console.warn('[audit] dropped event with unsafe detail', { type: event.type });
                return;
            }
            this.buffer.push(event);
            if (this.buffer.length > this.maxBuffer) this.buffer.shift();
            if (this.sink) {
              try {
                    const r = this.sink(event);
                    if (r instanceof Promise) r.catch(() => undefined);
                } catch {
                  // Swallow: audit must never interrupt the session.
                }
            }
        }


        /** Recent events for display (newest last). */
        recent(limit = 50): SessionAuditEvent[] {
            return this.buffer.slice(-limit);
        }
   }

   import { describe, it, expect } from 'vitest';
   import { InputRateLimiter } from '../../apps/desktop/src/renderer/src/services/inputRateLimiter';


   describe('InputRateLimiter', () => {
     it('allows a burst up to capacity then blocks', () => {
         // discrete rate 10/s, burst 2 => capacity 20
         const rl = new InputRateLimiter({ discreteRatePerSec: 10, burstFactor: 2 }, 0);
         let allowed = 0;
         for (let i = 0; i < 25; i++) if (rl.allow('key', 0)) allowed += 1;
         expect(allowed).toBe(20); // capacity
       });


       it('refills over time', () => {
         const rl = new InputRateLimiter({ discreteRatePerSec: 10, burstFactor: 1 }, 0);
         for (let i = 0; i < 10; i++) rl.allow('key', 0); // drain
         expect(rl.allow('key', 0)).toBe(false);
         // 1 second later => 10 tokens refilled
         expect(rl.allow('key', 1000)).toBe(true);
       });


       it('gives high-frequency kinds a larger bucket', () => {
         const rl = new InputRateLimiter({ discreteRatePerSec: 5, highFreqRatePerSec: 100, burstFactor: 1 }, 0);
         let moves = 0;
         for (let i = 0; i < 100; i++) if (rl.allow('mouse_move', 0)) moves += 1;
         expect(moves).toBe(100);
         // discrete bucket is independent + smaller
         let keys = 0;
         for (let i = 0; i < 100; i++) if (rl.allow('key', 0)) keys += 1;
         expect(keys).toBe(5);
     });
   });
