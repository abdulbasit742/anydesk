  import type { EntitlementSet, FeatureKey, GateResult, PlanTier } from './types';
  import { getPlan } from './planLimits';


  /**
   * Pure feature gating shared by server enforcement + client UI gating so they
      * never drift. Fails CLOSED: an unknown/fail-closed entitlement denies the
      * feature unless the safe fallback plan explicitly includes it.
   */
  export function resolveEntitlements(
       tier: PlanTier,
       failClosed = false,
  ): EntitlementSet {
    const plan = getPlan(tier);
       return {
         tier: plan.tier,
         features: [...plan.features],
         limits: { ...plan.limits },
         resolvedAt: new Date().toISOString(),
         failClosed,
       };
  }


  export function checkFeature(
       entitlements: EntitlementSet,
       feature: FeatureKey,
  ): GateResult {
    if (entitlements.failClosed) {
           // Even fail-closed, honor features in the safe fallback plan only.
           const allowed = entitlements.features.includes(feature);
           return { feature, allowed, reason: allowed ? 'allowed' : 'fail_closed' };
       }
       const allowed = entitlements.features.includes(feature);
       return { feature, allowed, reason: allowed ? 'allowed' : 'not_in_plan' };
  }


  /** Convenience: which of a list of features are blocked. */
  export function blockedFeatures(
       entitlements: EntitlementSet,
       features: FeatureKey[],
  ): FeatureKey[] {
    return features.filter((f) => !checkFeature(entitlements, f).allowed);
  }


packages/shared/src/entitlements/usageWindow.ts         — SAFE_DIRECT_COPY
  import type { UsageCounterDto, UsageMetric } from './types';


  /**
      * Pure usage-window math. A window is a UTC calendar month by default. Shared so
      * the API counts and the UI displays the SAME window boundaries.
   */
  export function currentMonthlyWindow(now: Date = new Date()): { start: Date; end: Date } {
       const start = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), 1));
       const end = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth() + 1, 1));
       return { start, end };
  }


  /** A stable key for a metric + window, used as the UsageCounter primary key. */
  export function usageKey(metric: UsageMetric, windowStart: Date): string {
    return `${metric}:${windowStart.toISOString().slice(0, 10)}`;
  }


  /** Build a counter DTO + compute exceeded. */
  export function toCounterDto(
       metric: UsageMetric,
       used: number,
       limit: number | null,
       window: { start: Date; end: Date },
  ): UsageCounterDto {
    return {
           metric,
           used,
           limit,
           windowStart: window.start.toISOString(),
           windowEnd: window.end.toISOString(),
           exceeded: limit !== null && used >= limit,
       };
  }


  /**
      * Would incrementing by `amount` exceed the limit? Fails CLOSED: a null/unknown
      * limit when `failClosed` is set blocks the increment.
   */
  export function wouldExceed(
       used: number,
       limit: number | null,
       amount: number,
       failClosed = false,
  ): boolean {
    if (limit === null) return failClosed ? true : false;
       return used + amount > limit;
  }

  import { createHmac, timingSafeEqual } from 'node:crypto';


  /**
      * ISOLATED Stripe signature verification. This is the ONLY place signature logic
      * lives, so it can be swapped for the real `stripe.webhooks.constructEvent` later
      * without touching the route/service.
      *
      * Dev mode: if no signing secret is configured, verification is SKIPPED and the
      * event is treated as unverified (verified=false) but still parsed — so the app
      * runs without real Stripe keys. It NEVER fabricates a verified=true result
      * without a real secret + matching signature.
   */
  export interface StripeVerifyResult {
       verified: boolean;
       /** Parsed event id + type (best-effort). Never includes the signature. */
       eventId: string | null;
       type: string | null;
  }


  export function verifyStripeSignature(
    rawBody: string,
       signatureHeader: string | undefined,
       signingSecret: string | undefined,
  ): StripeVerifyResult {
    const parsed = safeParse(rawBody);
       const eventId = parsed?.id ?? null;
       const type = parsed?.type ?? null;

       // Dev: no secret configured -> cannot verify. Treat as unverified, not fatal.
       if (!signingSecret) {
         return { verified: false, eventId, type };
       }
       if (!signatureHeader) {
           return { verified: false, eventId, type };
       }


       // Minimal Stripe-style "t=...,v1=..." parsing + HMAC-SHA256 check.
       const { t, v1 } = parseSigHeader(signatureHeader);
       if (!t || !v1) return { verified: false, eventId, type };
       const expected = createHmac('sha256', signingSecret)
         .update(`${t}.${rawBody}`)
           .digest('hex');


       const verified = safeEqualHex(expected, v1);
       return { verified, eventId, type };
  }

  function parseSigHeader(header: string): { t: string | null; v1: string | null } {
    let t: string | null = null;
       let v1: string | null = null;
       for (const part of header.split(',')) {
           const [k, val] = part.split('=');
           if (k === 't') t = val;
           if (k === 'v1') v1 = val;
       }
       return { t, v1 };
  }

  function safeEqualHex(a: string, b: string): boolean {
       try {
         const ba = Buffer.from(a, 'hex');
           const bb = Buffer.from(b, 'hex');
           return ba.length === bb.length && timingSafeEqual(ba, bb);
       } catch {
         return false;
       }
  }


  function safeParse(raw: string): { id?: string; type?: string } | null {
       try {
         return JSON.parse(raw);
       } catch {
         return null;
       }
  }

  import {
    currentMonthlyWindow,
    toCounterDto,
    wouldExceed,
    type UsageCounterDto,
    type UsageMetric,
  } from '@remotedesk/shared';
  import { prisma } from '../db/prisma';

  /**
   * Usage counter helpers. Increments are atomic via Prisma upsert + BigInt add.
   * Limits are passed in by the caller (resolved from entitlements) so this stays
   * a pure persistence concern.
   */
  export const usageCounters = {
    async read(
      subscriptionId: string,
      metric: UsageMetric,
      limit: number | null,
    ): Promise<UsageCounterDto> {
      const window = currentMonthlyWindow();
      const row = await prisma.usageCounter.findUnique({
        where: {
            subscriptionId_metric_windowStart: {
              subscriptionId,
               metric,
               windowStart: window.start,
            },
          },
      });
      const used = row ? Number(row.used) : 0;
      return toCounterDto(metric, used, limit, window);
    },


    /**
     * Increment with a fail-closed check. Returns the updated counter. Throws via
     * the caller if it would exceed (caller decides). Here we just enforce + write.
     */
    async increment(
      subscriptionId: string,
      metric: UsageMetric,
      amount: number,
      limit: number | null,
      failClosed: boolean,
    ): Promise<{ counter: UsageCounterDto; exceeded: boolean }> {
      const window = currentMonthlyWindow();
      const current = await this.read(subscriptionId, metric, limit);
      if (wouldExceed(current.used, limit, amount, failClosed)) {
        return { counter: current, exceeded: true };
      }
      const row = await prisma.usageCounter.upsert({
          where: {
            subscriptionId_metric_windowStart: {
               subscriptionId,
               metric,
                  windowStart: window.start,
                },
             },
             create: {
               subscriptionId,
                metric,
                windowStart: window.start,
                windowEnd: window.end,
                used: BigInt(amount),
             },
             update: { used: { increment: BigInt(amount) } },
           });
           return {
             counter: toCounterDto(metric, Number(row.used), limit, window),
             exceeded: false,
           };
       },
  };

  import { prisma } from '../db/prisma';


  /**
      * Billing audit writer. Detail is content-light and MUST NOT contain secrets,
      * signatures, or keys. A small allow-list scrub strips anything that looks
      * sensitive before persistence (defense in depth).
      */
  const FORBIDDEN = [/secret/i, /signature/i, /authorization/i, /api[-_]?key/i, /token/i, /password/i];


  function scrub(detail: Record<string, unknown> | undefined): Record<string, unknown> | undefined {
    if (!detail) return undefined;
       const out: Record<string, unknown> = {};
       for (const [k, v] of Object.entries(detail)) {
           out[k] = FORBIDDEN.some((re) => re.test(k)) ? '[REDACTED]' : v;
       }
       return out;
  }


  export const billingAudit = {
       async record(input: {
         subscriptionId?: string | null;
           userId?: string | null;
           action: string;
         detail?: Record<string, unknown>;
       }): Promise<void> {
           await prisma.billingAuditEvent.create({
             data: {
                subscriptionId: input.subscriptionId ?? null,
                userId: input.userId ?? null,
                action: input.action,
                detail: (scrub(input.detail) as unknown as object) ?? undefined,
             },
           });
       },
  };

           return false; // unique violation -> duplicate delivery

         await prisma.stripeWebhookEvent.update({
           where: { stripeEventId },

         return Promise.all(
           metrics.map((m) => usageCounters.read(subscriptionId, m, entitlements.limits[m])),
         );
       },


       async increment(
         subscriptionId: string,
         metric: UsageMetric,
         amount: number,
       ): Promise<UsageCounterDto> {
         const entitlements = await this.forSubscription(subscriptionId);
         const { counter, exceeded } = await usageCounters.increment(
            subscriptionId,
            metric,
            amount,
            entitlements.limits[metric],
            entitlements.failClosed,
         );
         await billingAudit.record({
           subscriptionId,
            action: 'usage.incremented',
            detail: { metric, amount, exceeded },
         });
         if (exceeded) throw new AppError(ErrorCodes.USAGE_LIMIT_EXCEEDED);
         return counter;
       },
  };

  import { verifyStripeSignature } from '../../lib/stripeSignature';
  import { billingAudit } from '../../lib/billingAudit';
  import { billingRepository } from '../../repositories/billingRepository';


  /**
   * Handle a raw Stripe webhook. Verification is delegated to the isolated stub.
   * Failed verification is AUDIT LOGGED (without the signature/secret) and the
   * request is rejected. No real keys are required to run in dev — unverified
   * events are recorded as verified=false and not processed.
   */
  export const stripeWebhookService = {
       async handle(input: {
         rawBody: string;
         signatureHeader: string | undefined;
         signingSecret: string | undefined;
       }): Promise<{ accepted: boolean; verified: boolean }> {
         const result = verifyStripeSignature(
             input.rawBody,
             input.signatureHeader,
             input.signingSecret,
         );


         if (!result.eventId) {
             await billingAudit.record({
               action: 'stripe.webhook.unparseable',
               detail: { verified: result.verified },
             });
             return { accepted: false, verified: result.verified };
         }


         if (!result.verified) {
             // Never store the raw signature or secret — only the outcome.
             await billingAudit.record({
               action: 'stripe.webhook.verify_failed',
               detail: { eventId: result.eventId, type: result.type },
             });
             await billingRepository.recordWebhookEvent({
               stripeEventId: result.eventId,
               type: result.type ?? 'unknown',
               verified: false,
             });
             return { accepted: false, verified: false };
         }


         const isNew = await billingRepository.recordWebhookEvent({
             stripeEventId: result.eventId,
             type: result.type ?? 'unknown',
           verified: true,
         });
         if (isNew) {
           // Real plan-sync handlers wire here in a later pack (no fake completion now).
             await billingRepository.markProcessed(result.eventId);
             await billingAudit.record({
               action: 'stripe.webhook.received',
               detail: { eventId: result.eventId, type: result.type },
             });
         }
         return { accepted: true, verified: true };
       },
  };

  export class EntitlementsClient {

           private readonly baseUrl: string,
           private readonly authToken: string,

       async fetch(): Promise<EntitlementSet> {
         try {
               const res = await fetch(`${this.baseUrl}/api/billing/entitlements`, {
                 headers: { authorization: `Bearer ${this.authToken}` },
               });
               if (!res.ok) throw new Error(`entitlements fetch failed: ${res.status}`);
             return (await res.json()) as EntitlementSet;
           } catch {
               // Fail closed: most restrictive plan, flagged.
               return resolveEntitlements('free', true);
           }
       }
  }

  # Billing Hardening: Entitlements, Usage Limits & Audit-Safe Plan Enforcement


  A Stripe-**ready** entitlements + usage foundation. **This is NOT production-ready
  billing:** there is no live checkout, no portal, and the Stripe webhook verifier
  is an isolated stub. No real Stripe secret keys are required to run in dev, and
  no payment is ever faked as complete.


  ## Plans & features
  `PLAN_DEFINITIONS` in `@remotedesk/shared` is the single source of truth for tier
  features + numeric limits (free / pro / business / enterprise). Stripe product/price
  IDs are intentionally absent. `resolveEntitlements(tier)` and `checkFeature()` run
  identically on the API (enforcement) and desktop/web (UI gating).


  ## Fail-closed enforcement
  - Unknown/unresolvable tier → `resolveForTier` returns the **free** plan flagged
    `failClosed: true`.
  - `checkFeature` on a fail-closed set denies anything outside the safe fallback
    with reason `fail_closed`.
  - `wouldExceed(used, null, amount, failClosed=true)` blocks increments when the
    limit is unknown.
  - Desktop `EntitlementsClient.fetch()` returns the fail-closed set on ANY error,
    so gated UI (file transfer / clipboard / unattended) is disabled, not enabled.


  ## Endpoints
  - `GET /api/billing/entitlements` — resolved entitlement set (audited).
  - `GET /api/billing/usage` — usage counters for the current UTC-month window.
  - `POST /api/billing/usage/increment` — increment a metric; throws `USAGE_LIMIT_EXCEEDED` when over.
  - `POST /api/billing/stripe/webhook` — raw-body webhook; verification via the isolated stub.


  ## Stripe webhook (isolated stub)
  `lib/stripeSignature.ts` is the ONLY place signature logic lives. In dev (no
  `STRIPE_WEBHOOK_SECRET`) events are parsed but recorded `verified: false` and not
  processed. With a secret, it does a real HMAC-SHA256 check. **Failed verification
  is audit-logged (event id + type only — never the signature or secret)** and the
  request is rejected. Events are de-duped by `stripeEventId`.


  ## Audit safety
  `billingAudit.record()` scrubs any detail key matching secret/signature/authorization/
  api key/token/password before persistence. No secrets are ever written.


  ## Not done yet (see manifest knownLimitations)
  Live checkout, billing portal, and real plan-sync handlers are NOT wired. Native
  input execution remains disabled.
