   import type { Ed25519Verify } from './receiptSignature';

   /**
    * Ed25519 signature verification using the renderer's WebCrypto (SubtleCrypto).
    * This is the browser/Electron-renderer counterpart to the API's node:crypto
    * verifier — SAME algorithm, so a signature made server-side verifies here.
    * Provides the injectable Ed25519Verify the shared verifyReceiptSignature needs.
    *
    * Keys + signatures are base64url (matching the server). No private material.
    */
   export const webcryptoEd25519Verify: Ed25519Verify = async (
        publicKeyB64Url: string,
        message: string,
     signatureB64Url: string,
   ): Promise<boolean> => {
        try {
          const key = await crypto.subtle.importKey(
              'raw',
              fromB64url(publicKeyB64Url),
              { name: 'Ed25519' },
              false,
              ['verify'],
            );
            return await crypto.subtle.verify(
              { name: 'Ed25519' },
              key,
              fromB64url(signatureB64Url),
              new TextEncoder().encode(message),
            );
        } catch {
          return false; // unsupported/invalid => fail closed
        }
   };


   function fromB64url(s: string): Uint8Array {
        const b64 = s.replace(/-/g, '+').replace(/_/g, '/');
        const pad = b64.length % 4 ? '='.repeat(4 - (b64.length % 4)) : '';
        const bin = atob(b64 + pad);
        const out = new Uint8Array(bin.length);
        for (let i = 0; i < bin.length; i++) out[i] = bin.charCodeAt(i);
        return out;
   };



FILE: packages/shared/src/permissions/keyPinning.ts
   import type { SigningPublicKey } from './receiptSignature';

   /**
    * Trust-on-first-use pinning for the org's receipt signing key. On first sight we
    * PIN { keyId -> publicKey }. Later, the same keyId MUST present the same public
    * key, or it's a substitution (fail closed). Rotation introduces a NEW keyId,
    * which is pinned fresh — a legitimate rotation never reuses a keyId with a
    * different key.
    */
   export interface PinnedKey {
        keyId: string;
        publicKey: string;
        pinnedAt: string;
   }

   export type PinResult =
        | { status: 'pinned_new' }              // first time we saw this keyId
        | { status: 'match' }                   // keyId + publicKey match the pin
        | { status: 'mismatch' };               // SAME keyId, DIFFERENT key => substitution


   /**
    * Evaluate an incoming public key against the existing pins. Pure: returns the
       * result + the (possibly updated) pin set so the caller can persist it.
       */
   export function evaluatePin(
     pins: Record<string, PinnedKey>,
        incoming: SigningPublicKey,
        now: Date = new Date(),
   ): { result: PinResult; pins: Record<string, PinnedKey> } {
     const existing = pins[incoming.keyId];
        if (!existing) {
          return {
             result: { status: 'pinned_new' },
             pins: { ...pins, [incoming.keyId]: { keyId: incoming.keyId, publicKey: incoming.publicKey, pinnedAt:
   now.toISOString() } },
       };
        }
        if (existing.publicKey === incoming.publicKey) {
            return { result: { status: 'match' }, pins };
        }
        return { result: { status: 'mismatch' }, pins }; // do NOT overwrite a mismatching pin
   }


   /** A pin result is acceptable for trusting a signature only if new or matching. */
   export function isPinAcceptable(result: PinResult): boolean {
     return result.status === 'pinned_new' || result.status === 'match';
   }

   import {
     evaluatePin,
        isPinAcceptable,
        type PinResult,
        type PinnedKey,
        type SigningPublicKey,
   } from '@remotedesk/shared';


   /**
       * Persists pinned signing public keys in localStorage and evaluates incoming keys
       * against them. Fail-closed: a same-keyId/different-key mismatch is never
       * auto-accepted and never overwrites the existing pin. Rotation (new keyId) pins
       * fresh automatically.
    */
   const PIN_KEY = 'remotedesk.signingKeyPins';


   export class SigningKeyPinStore {
        private load(): Record<string, PinnedKey> {
          try {
              return JSON.parse(localStorage.getItem(PIN_KEY) ?? '{}') as Record<string, PinnedKey>;
            } catch {
                return {};
            }
        }


        private save(pins: Record<string, PinnedKey>): void {
          try {
              localStorage.setItem(PIN_KEY, JSON.stringify(pins));
            } catch {
                /* storage unavailable: pins are best-effort this session */
            }
        }


        /** Evaluate + persist (only on new/match; a mismatch leaves the pin untouched). */
        check(incoming: SigningPublicKey): PinResult {
            const pins = this.load();
            const { result, pins: next } = evaluatePin(pins, incoming);
            if (isPinAcceptable(result)) this.save(next);
            return result;
        }


        /** Get the pinned public key for a keyId, if any. */
        pinned(keyId: string): PinnedKey | null {
            return this.load()[keyId] ?? null;
        }


        /** Operator escape hatch: forget a pin (e.g. after a verified rotation incident). */
        forget(keyId: string): void {
          const pins = this.load();
            delete pins[keyId];
            this.save(pins);
        }
   }

/**
    * Full desktop-side verification of a signed consent receipt:
    *   1. hash-chain intact (Pack 22 verifyReceipt, with WebCrypto sha256)
    *        2. signing key pin acceptable (Pack 28 pinning) — detects key substitution
    *        3. Ed25519 signature valid over the canonical preimage (Pack 26 verifier)
    * All three must pass. Fail-closed: any failure => verified:false with a reason.
    * No server call, no private material.
 */
export interface DesktopVerifyResult {
        verified: boolean;
        chainValid: boolean;
        signatureValid: boolean;
        pin: PinResult | 'no_signature';
        reason: string | null;
}


export class ReceiptVerifyClient {
        constructor(private readonly pins = new SigningKeyPinStore()) {}

        async verify(orgId: string, receipt: ConsentReceipt, publicKey: SigningPublicKey | null): Promise<DesktopVerifyResult>
{
            // 1. Chain integrity (no signature needed).
            const chain = await verifyReceipt(receipt, sha256Hex);
            if (!chain.valid) {
              return { verified: false, chainValid: false, signatureValid: false, pin: 'no_signature', reason: 'chain_invalid' };
            }


            // No signature on the receipt yet (server didn't sign) => chain-only, unverified sig.
            if (!receipt.signature || !publicKey) {
                return { verified: false, chainValid: true, signatureValid: false, pin: 'no_signature', reason: 'unsigned' };
            }


            // 2. Pin check (TOFU). Mismatch => possible key substitution.
            const pin = this.pins.check(publicKey);
            if (!isPinAcceptable(pin)) {
                return { verified: false, chainValid: true, signatureValid: false, pin, reason: 'key_pin_mismatch' };
            }


            // 3. Signature check against the (pinned) public key.
    const signatureValid = await verifyReceiptSignature(orgId, receipt, receipt.signature, publicKey,
webcryptoEd25519Verify);
            return {
              verified: signatureValid,
                chainValid: true,
                signatureValid,
                pin,
                reason: signatureValid ? null : 'signature_invalid',
            };
        }
   }


   async function sha256Hex(input: string): Promise<string> {
        const digest = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(input));
        return Array.from(new Uint8Array(digest), (b) => b.toString(16).padStart(2, '0')).join('');
   }

  returns a signed receipt (Pack 26), the host independently:

rotation from Pack 27) pins fresh while old keyIds keep their pins for old receipts, and a mismatching pin is
never overwritten. The desktop never touches private material, just the published public key.
