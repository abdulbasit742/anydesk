/**
 * One-call, browser-friendly receipt verification for third parties. Combines the
    * Pack 22 chain check + the Pack 26 signature check using WebCrypto, so an auditor
    * page can verify a downloaded receipt with a single function and no server crypto.
    * Fail-closed: any failure yields a precise, non-sensitive reason.
    */
export type VerifyBundleReason =
  | 'ok'
     | 'malformed_receipt'
     | 'chain_invalid'
     | 'unsigned'
     | 'no_public_key'
     | 'key_id_mismatch'
     | 'signature_invalid';


export interface VerifyBundleResult {
     ok: boolean;
     chainValid: boolean;
     signatureValid: boolean;
     reason: VerifyBundleReason;
     /** First broken timeline seq when the chain fails, else null. */
     firstBrokenSeq: number | null;
     /** Echoed non-sensitive facts for display. */
     facts: {
       receiptId: string | null;
       deviceId: string | null;
       rootHash: string | null;
       issuedAt: string | null;
       keyId: string | null;
     };
}


/** SHA-256 hex via WebCrypto (auditor browser). */
async function sha256Hex(input: string): Promise<string> {
     const digest = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(input));
     return Array.from(new Uint8Array(digest), (b) => b.toString(16).padStart(2, '0')).join('');
}

function facts(r: Partial<ConsentReceipt> | null) {
  return {
       receiptId: r?.receiptId ?? null,
       deviceId: r?.summary?.deviceId ?? null,
       rootHash: r?.rootHash ?? null,
       issuedAt: r?.issuedAt ?? null,
       keyId: r?.signature?.keyId ?? null,
     };
}

/** Basic shape guard before any crypto. */
export function isReceiptShaped(value: unknown): value is ConsentReceipt {
     if (!value || typeof value !== 'object') return false;
     const r = value as Record<string, unknown>;
     return (
       typeof r.receiptId === 'string' &&
       typeof r.rootHash === 'string' &&
       typeof r.issuedAt === 'string' &&
         Array.isArray(r.lineItems) &&
         typeof r.summary === 'object'
       );
  }


  /**
   * Verify a receipt against an org id + the org's published public key. The public
      * key is fetched by the page and passed in; the crypto runs locally.
      */
  export async function verifyReceiptBundle(
    orgId: string,
       receipt: unknown,
       publicKey: SigningPublicKey | null,
  ): Promise<VerifyBundleResult> {
    if (!isReceiptShaped(receipt)) {
      return { ok: false, chainValid: false, signatureValid: false, reason: 'malformed_receipt', firstBrokenSeq: null,
  facts: facts(null) };
       }
       const r = receipt as ConsentReceipt;

       const chain = await verifyReceipt(r, sha256Hex);
       if (!chain.valid) {
         return { ok: false, chainValid: false, signatureValid: false, reason: 'chain_invalid', firstBrokenSeq:
  chain.firstBrokenSeq, facts: facts(r) };
    }
       if (!r.signature) {
         return { ok: false, chainValid: true, signatureValid: false, reason: 'unsigned', firstBrokenSeq: null, facts:
  facts(r) };
    }
       if (!publicKey) {
         return { ok: false, chainValid: true, signatureValid: false, reason: 'no_public_key', firstBrokenSeq: null, facts:
  facts(r) };
    }
       if (publicKey.keyId !== r.signature.keyId) {
         return { ok: false, chainValid: true, signatureValid: false, reason: 'key_id_mismatch', firstBrokenSeq: null, facts:
  facts(r) };
    }


       const signatureValid = await verifyReceiptSignature(orgId, r, r.signature, publicKey, webcryptoEd25519Verify);
       return {
         ok: signatureValid,
         chainValid: true,
         signatureValid,
         reason: signatureValid ? 'ok' : 'signature_invalid',
         firstBrokenSeq: null,
         facts: facts(r),
       };
  }

     /** Fetch a published public key by its keyId (rotation-safe: old keys resolvable). */
     .get('/signing-key/:keyId', h(async (req, res) => {
       const pub = await signingKeyRepository.publicKeyById(req.params.keyId);
         if (!pub) {
           res.status(404).json({ error: 'unknown_key' });
             return;
         }
         // Public material only — keyId, publicKey, alg, timestamps.
         res.json(pub);
     }))
     /** Lightweight liveness for the verify page. */
     .get('/health', h(async (_req, res) => {
       res.json({ ok: true });
     }));

    * Public verify page API. Fetches ONLY the published public key by keyId. Uses a

   const BASE = process.env.NEXT_PUBLIC_VERIFY_API_BASE ?? '';


   export const verifyApi = {
     async publicKey(keyId: string): Promise<SigningPublicKey | null> {
          try {
            const res = await fetch(`${BASE}/verify/signing-key/${encodeURIComponent(keyId)}`);
              if (!res.ok) return null;
              return (await res.json()) as SigningPublicKey;
          } catch {
            return null;
          }
        },
   };

              setError('That file is not valid JSON.');
          }
        }, [onReceipt]);


        return (
          <div
              className={`${styles.dropzone} ${dragging ? styles.dropzoneActive : ''}`}
              onDragOver={(e) => { e.preventDefault(); setDragging(true); }}
              onDragLeave={() => setDragging(false)}

   handleFile(f); }}
       >
              <p>Drop a <code>consent-receipt-*.json</code> here, or</p>
              <label className={styles.picker}>
           <input type="file" accept="application/json,.json" hidden onChange={(e) => { const f = e.target.files?.[0]; if
   (f) void handleFile(f); }} />
                choose a file
              </label>
              <p className={styles.privacy}>The file is verified in your browser. Nothing is uploaded.</p>
            {error && <div className={styles.error}>{error}</div>}
          </div>
        );
   }

   const REASON_TEXT: Record<string, string> = {
     ok: 'Receipt is authentic: hash chain intact and signature valid.',
        malformed_receipt: 'This file is not a RemoteDesk consent receipt.',
        chain_invalid: 'The receipt timeline has been altered (hash chain broken).',
        unsigned: 'The receipt has no signature (chain is intact, but not signed).',
        no_public_key: 'Could not fetch the signing public key for this receipt.',
        key_id_mismatch: 'The fetched key does not match the receipt signature key id.',
        signature_invalid: 'The signature does not match (possible tampering or wrong org).',

   /**
    * Public, no-account receipt verification page. An auditor supplies the orgId the
       * receipt belongs to, drops the JSON, and the page verifies it locally against the
       * org's published public key.
    */
   export default function VerifyPage() {
        const [orgId, setOrgId] = useState('');
        const [result, setResult] = useState<VerifyBundleResult | null>(null);
        const [busy, setBusy] = useState(false);
        const [fileName, setFileName] = useState<string | null>(null);


        const onReceipt = async (parsed: unknown, name: string) => {
          setFileName(name);
          setBusy(true);
          setResult(null);

              const keyId = (parsed as ConsentReceipt)?.signature?.keyId ?? null;
              const publicKey = keyId ? await verifyApi.publicKey(keyId) : null;
            setResult(await verifyReceiptBundle(orgId.trim(), parsed, publicKey));
          } finally {
              setBusy(false);
          }
        };


        return (
          <main className={styles.page}>
              <header className={styles.header}>
                <h1>Verify a consent receipt</h1>
                <p className={styles.sub}>Independently confirm a RemoteDesk session receipt. Runs entirely in your browser.</p>
              </header>
            <label className={styles.field}>
              <span>Organization ID (from the receipt holder)</span>
              <input value={orgId} onChange={(e) => setOrgId(e.target.value)} placeholder="org_…" />
            </label>


            <ReceiptDropzone onReceipt={onReceipt} />


            {busy && <p className={styles.muted}>Verifying…</p>}


            {result && (
              <section className={styles.result} data-ok={result.ok}>
                <h2>{result.ok ? '✓ Authentic' : '✗ Not verified'}</h2>
                <p>{REASON_TEXT[result.reason] ?? result.reason}</p>
                {result.firstBrokenSeq !== null && <p className={styles.muted}>First broken timeline entry: #
