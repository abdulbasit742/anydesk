import type { SigningPublicKey } from './receiptSignature';


/**
    * Rotation policy: how often to rotate the receipt signing key and how long a
    * just-retired key remains flagged as overlapping. Pure helpers decide when
    * rotation is due and whether a retired key is still within grace.
    *
    * Overlap semantics: a new active key is minted BEFORE the old one is retired, so
    * there is never a moment without a valid signing key. After retirement, the old
    * key still VERIFIES forever (never deleted) but is flagged "in grace" for a
    * window so clients/auditors can treat a recent rotation leniently.
 */
export interface RotationPolicy {
     maxKeyAgeDays: number;
     overlapDays: number;
     enabled: boolean;
}

export const DEFAULT_ROTATION_POLICY: RotationPolicy = {
     maxKeyAgeDays: 90,
     overlapDays: 7,
     enabled: false, // opt-in: rotation is off until an admin enables it
};


/** Clamp + validate a policy from untrusted input. */
export function normalizeRotationPolicy(p: Partial<RotationPolicy>): RotationPolicy {
  return {
       maxKeyAgeDays: clampInt(p.maxKeyAgeDays ?? DEFAULT_ROTATION_POLICY.maxKeyAgeDays, 1, 3650),
       overlapDays: clampInt(p.overlapDays ?? DEFAULT_ROTATION_POLICY.overlapDays, 0, 365),
       enabled: !!p.enabled,
     };
}


function clampInt(n: number, min: number, max: number): number {
  const v = Math.floor(Number.isFinite(n) ? n : min);
     return Math.min(max, Math.max(min, v));
}


/** Age of the active key in whole days. */
export function keyAgeDays(activeKey: SigningPublicKey, now: Date = new Date()): number {
  return Math.floor((now.getTime() - Date.parse(activeKey.createdAt)) / 86_400_000);
}


/**
 * Is rotation due? True only when enabled, an active key exists, and its age
    * exceeds maxKeyAgeDays. Fail-safe: no active key => due (so a key gets minted).
    */
export function isRotationDue(policy: RotationPolicy, activeKey: SigningPublicKey | null, now: Date = new Date()):
boolean {
     if (!policy.enabled) return false;
     if (!activeKey) return true;
     return keyAgeDays(activeKey, now) >= policy.maxKeyAgeDays;
}


/** Is a retired key still within its overlap/grace window? */
export function isWithinOverlap(policy: RotationPolicy, retiredKey: Pick<SigningPublicKey, 'retiredAt'>, now: Date = new
   Date()): boolean {
       if (!retiredKey.retiredAt) return false;
       const graceEnd = Date.parse(retiredKey.retiredAt) + policy.overlapDays * 86_400_000;
       return now.getTime() < graceEnd;
   }

   import { normalizeRotationPolicy, DEFAULT_ROTATION_POLICY, type RotationPolicy } from '@remotedesk/shared';
   import { prisma } from '../db/prisma';


   export const rotationPolicyRepository = {
       async get(orgId: string): Promise<RotationPolicy & { lastRunAt: string | null }> {
         const row = await prisma.keyRotationPolicy.findUnique({ where: { orgId } });
         if (!row) return { ...DEFAULT_ROTATION_POLICY, lastRunAt: null };
         return {
              maxKeyAgeDays: row.maxKeyAgeDays,
              overlapDays: row.overlapDays,
              enabled: row.enabled,
              lastRunAt: row.lastRunAt ? row.lastRunAt.toISOString() : null,
          };
        },


        async set(orgId: string, input: Partial<RotationPolicy>): Promise<RotationPolicy> {
          const policy = normalizeRotationPolicy(input);
          await prisma.keyRotationPolicy.upsert({ where: { orgId }, create: { orgId, ...policy }, update: policy });
          return policy;
        },

        async markRun(orgId: string, at: Date = new Date()): Promise<void> {
          await prisma.keyRotationPolicy.updateMany({ where: { orgId }, data: { lastRunAt: at } });
        },


        async allEnabled(): Promise<Array<{ orgId: string; policy: RotationPolicy }>> {
          const rows = await prisma.keyRotationPolicy.findMany({ where: { enabled: true, orgId: { not: null } } });
       return rows.map((r) => ({ orgId: r.orgId as string, policy: { maxKeyAgeDays: r.maxKeyAgeDays, overlapDays:
   r.overlapDays, enabled: r.enabled } }));
        },
   };



FILE: apps/api/src/services/admin/scheduledRotation.service.ts
   import { isRotationDue, deriveRotationState, type RotationPolicy, type SigningPublicKey } from '@remotedesk/shared';
   import { signingKeyRepository } from '../../repositories/signingKeyRepository';
   import { rotationPolicyRepository } from '../../repositories/rotationPolicyRepository';
   import { deviceAdminAudit } from '../../lib/deviceAdminAudit';


   /**
    * Decides + performs scheduled rotation for one org. IDEMPOTENT: it only rotates
    * when the policy says rotation is due, so it's safe to call repeatedly (cron,
    * boot, manual run-now). Overlap is inherent: Pack 27 rotate() mints the new
    * active key THEN retires the old, so signing never has a gap. Retired keys are
    * kept (never deleted), so historical receipts always verify.
    */
   export const scheduledRotationService = {
     async runForOrg(orgId: string, policy?: RotationPolicy, now: Date = new Date()): Promise<{ rotated: boolean; newKeyId:
   string | null }> {
       const effective = policy ?? (await rotationPolicyRepository.get(orgId));
          const keys: SigningPublicKey[] = await (signingKeyRepository as any).listPublic(orgId);
          const state = deriveRotationState(keys);
          const activeKey = keys.find((k) => k.keyId === state.activeKeyId) ?? null;


          if (!isRotationDue(effective, activeKey, now)) {
            await rotationPolicyRepository.markRun(orgId, now);
              return { rotated: false, newKeyId: null };
          }


          const newKeyId: string = await (signingKeyRepository as any).rotate(orgId);
          await rotationPolicyRepository.markRun(orgId, now);
          await deviceAdminAudit.record({
              orgId, actorUserId: null, deviceId: 'signing-key',
              action: 'device.trust.changed',
                metadata: { event: 'signing_key_rotated', mode: 'scheduled', newKeyId, ageDays: String(activeKey ?
   Math.floor((now.getTime() - Date.parse(activeKey.createdAt)) / 86_400_000) : -1) },
       });
            return { rotated: true, newKeyId };
        },


        async runAll(now: Date = new Date()): Promise<Array<{ orgId: string; rotated: boolean; newKeyId: string | null }>> {
            const orgs = await rotationPolicyRepository.allEnabled();
            const out: Array<{ orgId: string; rotated: boolean; newKeyId: string | null }> = [];
            for (const { orgId, policy } of orgs) {
              try { out.push({ orgId, ...(await this.runForOrg(orgId, policy, now)) }); }
                catch { out.push({ orgId, rotated: false, newKeyId: null }); } // one org's failure never blocks the rest
            }
            return out;
        },
   };



FILE: apps/api/src/jobs/rotationRunner.ts
   import { scheduledRotationService } from '../services/admin/scheduledRotation.service';


   /**
       * Lightweight in-process scheduler. Calls runAll on an interval. In production
       * prefer an external cron / job queue hitting POST /run-now; this is a safe
       * default. Idempotent runs mean overlapping ticks are harmless.
       */
   let timer: ReturnType<typeof setInterval> | null = null;


   export function startRotationRunner(intervalMs = 6 * 60 * 60 * 1000): void {
     if (timer) return;
        void safeRun();
        timer = setInterval(() => void safeRun(), intervalMs);
   }


   export function stopRotationRunner(): void {
     if (timer) clearInterval(timer);
        timer = null;
   }


   async function safeRun(): Promise<void> {
        try {
          const summary = await scheduledRotationService.runAll();
            const rotated = summary.filter((s) => s.rotated).length;
            if (rotated > 0) console.info('[rotation] scheduled rotation complete', { orgsRotated: rotated });
        } catch (err) {
          console.warn('[rotation] runner error', { reason: err instanceof Error ? err.message : 'unknown' });
        }
   }

   'use client';


   import React, { useEffect, useState } from 'react';
   import type { RotationPolicy } from '@remotedesk/shared';
   import { api } from '../../../src/api/apiClient';


   type PolicyWithRun = RotationPolicy & { lastRunAt: string | null };


   /**
    * Admin panel to configure scheduled key rotation: enable, max key age, overlap
    * window, and a manual rotate-now trigger. No key material shown.
    */
   export function RotationPolicyPanel({ canManage }: { canManage: boolean }) {
     const [policy, setPolicy] = useState<PolicyWithRun | null>(null);
        const [busy, setBusy] = useState(false);
        const [msg, setMsg] = useState<string | null>(null);


        const load = async () => setPolicy(await api.get<PolicyWithRun>('/devices/receipts/rotation-policy'));
        useEffect(() => { void load(); }, []);


        const save = async () => {
          if (!policy) return;
         setBusy(true); setMsg(null);
         try {
         await api.put('/devices/receipts/rotation-policy', { maxKeyAgeDays: policy.maxKeyAgeDays, overlapDays:
   policy.overlapDays, enabled: policy.enabled });
           setMsg('Saved.');
        } finally { setBusy(false); }
      };


      const runNow = async () => {
        setBusy(true); setMsg(null);
        try {
          const r = await api.post<{ rotated: boolean; newKeyId: string | null }>('/devices/receipts/rotation-policy/run-
  now', {});
        setMsg(r.rotated ? `Rotated. New key: ${r.newKeyId}` : 'Not due yet — no rotation needed.');
          await load();
        } finally { setBusy(false); }
      };


      if (!policy) return <section className="rotation-policy-panel"><p>Loading rotation policy…</p></section>;


      return (
        <section className="rotation-policy-panel">
           <h2>Automatic key rotation</h2>
           <label>
          <input type="checkbox" disabled={!canManage} checked={policy.enabled} onChange={(e) => setPolicy({ ...policy,
  enabled: e.target.checked })} />
             Enable scheduled rotation
           </label>
           <div className="row">
             <label>Rotate after (days)
            <input type="number" min={1} max={3650} disabled={!canManage} value={policy.maxKeyAgeDays} onChange={(e) =>
  setPolicy({ ...policy, maxKeyAgeDays: Number(e.target.value) })} />
             </label>
             <label>Overlap window (days)
            <input type="number" min={0} max={365} disabled={!canManage} value={policy.overlapDays} onChange={(e) =>
  setPolicy({ ...policy, overlapDays: Number(e.target.value) })} />
             </label>
           </div>
           <p className="muted">Last run: {policy.lastRunAt ? new Date(policy.lastRunAt).toLocaleString() : 'never'}</p>
           {canManage && (
             <div className="row">
               <button type="button" disabled={busy} onClick={save}>Save policy</button>
               <button type="button" disabled={busy} onClick={runNow}>Rotate now</button>
             </div>
           )}
           {msg && <p className="muted">{msg}</p>}
        <p className="muted">A new key is created before the old one retires, so signing never stops. Retired keys keep
  verifying old receipts.</p>
        </section>
      );
  }

Pack 30 done: 12 files (8 created, 4 modified). Key rotation is now automatic and gap-free. A rotation policy
(max key age + overlap window + enable switch) drives an idempotent scheduled runner that rotates only when
the active key is older than the configured age, and a just-retired key stays flagged "in overlap" through a grace
window. Admins configure it all from the receipts page with a manual rotate-now button.


The clean bit: the overlap guarantee is structural, not timing-luck. Pack 27's rotate() mints the new active key
before retiring the old one, so there's never an instant without a valid signing key, and because retired keys are
never deleted, every historical receipt keeps verifying through Pack 28/29 regardless of how many rotations
have happened. The runner is idempotent: the due-check gates the actual rotate, so cron ticks, a boot kick, and
a manual run-now can all fire without ever double-rotating (a test proves a second immediate run is a no-op).
Policy input is clamped in shared, so neither the API nor the UI can persist nonsense.
