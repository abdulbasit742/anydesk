 Codex / Claude Code handoff. Direct continuation of Pack 23. Pack 23 left two flags: trust/grant mutations
 were auth-only (any org member could trust a device or mint a grant), and receipt POST trusted the desktop
 blindly. This slice adds (1) an admin RBAC guard so only owners/admins can change trust or create/revoke
 grants, and (2) server-side receipt verification, the API recomputes the consent receipt's hash chain before
 storing, rejects tampered/over-claiming receipts, and exposes a tamper-check endpoint. Build order: shared
 rbac/verify helpers → RBAC middleware → receipt verifier service → route guards → web gating → tests.
 Scope: lock down who can mutate trust/grants, and make stored receipts provably intact + content-free.
 Reuses Pack 6 RBAC ( can /permissions) where present. Server clamp still applies; host still owns live
 permissions. No secrets/payloads accepted into receipts. Fail-closed: unknown role => denied;
 unverifiable receipt => rejected.

   /**
       * Device-admin authorization. Reuses the org role model from Pack 6 (owner/admin/
       * member). Trust changes + grant create/revoke are admin-or-owner only. Pure +
       * fail-closed: an unknown/missing role denies.
       */
   export type OrgRoleLike = 'owner' | 'admin' | 'member' | string | null | undefined;

   export type DeviceAdminAction =
     | 'device.trust.set'
        | 'device.grant.create'
        | 'device.grant.revoke';


   /** Roles permitted to mutate device trust/grants. */
   const ADMIN_ROLES = new Set(['owner', 'admin']);


   export function canMutateDeviceAdmin(role: OrgRoleLike): boolean {
     return typeof role === 'string' && ADMIN_ROLES.has(role);
   }

   /** Per-action check (all device-admin actions currently require admin/owner). */
   export function canPerformDeviceAdminAction(role: OrgRoleLike, _action: DeviceAdminAction): boolean {
        return canMutateDeviceAdmin(role);
   }

   /** Read access: any member of the org may VIEW devices/receipts. */
   export function canViewDeviceAdmin(role: OrgRoleLike): boolean {
     return typeof role === 'string' && (ADMIN_ROLES.has(role) || role === 'member');
   }

   import { verifyReceipt, type ConsentReceipt } from './consentReceipt';
   import type { SessionSummary } from './sessionSummary';

   /**
    * Server-side receipt validation: structural (content-free) checks + hash-chain
       * verification. The chain check reuses the SAME verifyReceipt from Pack 22, so
       * the desktop and server agree on what "intact" means.
       */

   /** Keys that must NOT appear anywhere in a receipt (content-free invariant). */
   const FORBIDDEN_KEYS = [
        /payload/i, /keycode/i, /coords?/i, /clipboard/i, /filename/i, /filecontent/i,
        /token/i, /password/i, /secret/i, /authorization/i, /cookie/i, /\bip\b/i,
   ];


   /** Deep scan: returns the first forbidden key found, or null. */
   export function findForbiddenKey(value: unknown, path = ''): string | null {
        if (!value || typeof value !== 'object') return null;
        for (const [k, v] of Object.entries(value as Record<string, unknown>)) {
            if (FORBIDDEN_KEYS.some((re) => re.test(k))) return path ? `${path}.${k}` : k;
            const nested = findForbiddenKey(v, path ? `${path}.${k}` : k);
            if (nested) return nested;
       }
       return null;
  }


  export type ReceiptRejectReason =
       | 'bad_version'
       | 'missing_fields'
       | 'forbidden_content'
       | 'summary_mismatch'
       | 'chain_invalid';


  export interface ReceiptValidation {
    valid: boolean;
       reason: ReceiptRejectReason | null;
       firstBrokenSeq: number | null;
  }

  /** Structural checks that need no hashing. */
  export function structuralCheck(receipt: ConsentReceipt): ReceiptRejectReason | null {
       if (receipt.version !== 1) return 'bad_version';
       if (!receipt.receiptId || !receipt.rootHash || !receipt.summary || !Array.isArray(receipt.lineItems)) {
           return 'missing_fields';
       }
       if (findForbiddenKey(receipt)) return 'forbidden_content';
       if (!summaryConsistent(receipt.summary, receipt)) return 'summary_mismatch';
       return null;
  }


  /** Summary counts must not exceed what the line items can support (anti-overclaim). */
  function summaryConsistent(summary: SessionSummary, receipt: ConsentReceipt): boolean {
    const lineCount = receipt.lineItems.length;
       const c = summary.counts;
       const totalClaimed = c.inputBlocked + c.inputRateLimited + c.emergencyStops + c.controlGrants + c.controlRevokes;
       // Claimed events can't exceed the number of timeline lines.
       return totalClaimed <= lineCount;
  }


  /**
   * Full validation: structural first (cheap), then hash-chain via injected sha256.
      * Fail-closed: any structural failure short-circuits before hashing.
      */
  export async function validateReceipt(
    receipt: ConsentReceipt,
    sha256Hex: (input: string) => Promise<string>,
  ): Promise<ReceiptValidation> {
       const structural = structuralCheck(receipt);
       if (structural) return { valid: false, reason: structural, firstBrokenSeq: null };


       const chain = await verifyReceipt(receipt, sha256Hex);
       if (!chain.valid) return { valid: false, reason: 'chain_invalid', firstBrokenSeq: chain.firstBrokenSeq };


       return { valid: true, reason: null, firstBrokenSeq: null };
  }

   import type { Request, Response, NextFunction } from 'express';
   import { AppError, ErrorCodes, canMutateDeviceAdmin } from '@remotedesk/shared';
   import type { OrgRequest } from './orgContext.middleware';


   /**
    * Guards device-admin MUTATIONS (trust set, grant create/revoke). Relies on
       * orgContext having resolved the caller's org role (Pack 6). Fail-closed: a
       * missing/unknown role is denied.
    */
   export function requireDeviceAdmin(req: Request, _res: Response, next: NextFunction): void {
        const role = (req as OrgRequest & { orgRole?: string }).orgRole;
        if (!canMutateDeviceAdmin(role)) {
            return next(new AppError(ErrorCodes.PERMISSION_DENIED));
        }
        next();
   }

   import { createHash } from 'node:crypto';

   /**
    * Server SHA-256 (hex) matching the desktop's Web Crypto output. Used to verify
       * consent-receipt hash chains identically to how the desktop built them.
       */
   export async function nodeSha256Hex(input: string): Promise<string> {
     return createHash('sha256').update(input).digest('hex');
   }



FILE: apps/api/src/services/admin/receiptVerifier.service.ts
   import {
        AppError,
        ErrorCodes,
        validateReceipt,
        type ConsentReceipt,
     type ReceiptValidation,
   } from '@remotedesk/shared';
   import { nodeSha256Hex } from '../../lib/nodeSha256';
   import { consentReceiptRepository } from '../../repositories/consentReceiptRepository';


   /**
       * Verifies consent receipts server-side before storing, and re-verifies stored
    * ones on demand. A receipt that fails structural OR chain checks is rejected and
    * never stored. Verification reuses the shared validator with the node hash.
    */
   export const receiptVerifierService = {
     /** Validate + store. Throws if invalid (caller returns 400). */
        async storeVerified(orgId: string, receipt: ConsentReceipt): Promise<ReceiptValidation> {
          const result = await validateReceipt(receipt, nodeSha256Hex);
          if (!result.valid) throw new AppError(ErrorCodes.RECEIPT_INVALID);
          await consentReceiptRepository.store(orgId, receipt.summary.deviceId, receipt);
          return result;
        },

        /** Re-verify a previously stored receipt (tamper check at rest). */
        async verifyStored(orgId: string, receiptId: string): Promise<ReceiptValidation> {
          const receipt = await consentReceiptRepository.getJson(orgId, receiptId);
          if (!receipt) throw new AppError(ErrorCodes.RECEIPT_NOT_FOUND);
          return validateReceipt(receipt, nodeSha256Hex);
        },
   };

   import { describe, it, expect } from 'vitest';
   import { createHash } from 'node:crypto';
   import {
        validateReceipt,
        structuralCheck,
        findForbiddenKey,
        buildChain,
        makeAuditEvent,
        type ConsentReceipt,
     type SessionSummary,
   } from '@remotedesk/shared';


   const sha256Hex = async (i: string) => createHash('sha256').update(i).digest('hex');
   const events = [makeAuditEvent('control.granted'), makeAuditEvent('input.blocked'), makeAuditEvent('control.revoked')];


   const summary: SessionSummary = {
     deviceId: 'dev-1', startedAt: '2026-06-15T10:00:00.000Z', endedAt: '2026-06-15T10:05:00.000Z',
        durationSeconds: 300, capabilitiesUsed: { remoteInput: true, clipboardSync: false, fileTransfer: false },
     unattended: false, counts: { inputBlocked: 1, inputRateLimited: 0, emergencyStops: 0, controlGrants: 1, controlRevokes:
   1 },
        endReason: 'user_disconnect',
   };

   async function receipt(): Promise<ConsentReceipt> {
        const { lineItems, rootHash } = await buildChain(events, sha256Hex);
        return { receiptId: 'r1', summary, lineItems, rootHash, issuedAt: '2026-06-15T10:05:01.000Z', version: 1 };
   }


   describe('receiptVerification', () => {
     it('validates an intact, content-free receipt', async () => {
          expect(await validateReceipt(await receipt(), sha256Hex)).toEqual({ valid: true, reason: null, firstBrokenSeq: null
   });
        });


        it('rejects a wrong version', async () => {
          const r = await receipt(); (r as any).version = 2;
          expect((await validateReceipt(r, sha256Hex)).reason).toBe('bad_version');
        });


        it('rejects forbidden content keys anywhere', () => {
          expect(findForbiddenKey({ summary: { note: { clipboard: 'x' } } })).toBe('summary.note.clipboard');
          expect(findForbiddenKey({ a: { token: 1 } })).toBe('a.token');
          expect(findForbiddenKey({ a: { ok: 1 } })).toBeNull();
        });


        it('rejects a receipt that smuggles a forbidden key', async () => {
          const r = await receipt(); (r as any).leaked = { keycode: 65 };
          expect((await validateReceipt(r, sha256Hex)).reason).toBe('forbidden_content');
        });

        it('rejects over-claiming summary counts', async () => {
          const r = await receipt();
          r.summary = { ...r.summary, counts: { ...r.summary.counts, inputBlocked: 999 } };
          expect(structuralCheck(r)).toBe('summary_mismatch');
        });


        it('rejects a tampered chain', async () => {
          const r = await receipt();
          r.lineItems[1] = { ...r.lineItems[1], type: 'emergency_stop.engaged' };
          const v = await validateReceipt(r, sha256Hex);
          expect(v.reason).toBe('chain_invalid');
          expect(v.firstBrokenSeq).toBe(1);
     });
   });
