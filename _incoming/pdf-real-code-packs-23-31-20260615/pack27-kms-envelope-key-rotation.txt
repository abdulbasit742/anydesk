RemoteDesk · Pack 27 (API + Web) — KMS Envelope
Encryption for Signing Keys + Key Rotation (real code)
 Codex / Claude Code handoff. Direct continuation of Pack 26. Pack 26 flagged one production blocker: the
 Ed25519 private key sat in a plaintext DB column. This slice wraps it with envelope encryption (a KMS-held
 master key encrypts a per-key data key; the data key encrypts the private key), adds a pluggable KMS provider
 (with a safe dev fallback that is NOT production), and builds signing-key rotation so old receipts still verify
 under their original key.
 Scope: encrypt signing private keys at rest, decrypt only in-memory at sign time, rotate keys without breaking
 old signatures. Plaintext private keys never persisted, never logged, never returned. Dev fallback is
 clearly marked non-production. Rotation retires, never deletes.

   import type { SigningPublicKey } from './receiptSignature';

   /**
       * Key rotation model. Rotation creates a NEW active key and RETIRES the previous
       * one (never deletes), so signatures made under the old keyId still verify via
       * publicKeyById. Pure helpers for selecting the active key + describing state.
       */
   export interface KeyRotationState {
     activeKeyId: string | null;
        retiredKeyIds: string[];
        lastRotatedAt: string | null;
   }


   /** Build rotation state from a key list (newest active wins). */
   export function deriveRotationState(keys: SigningPublicKey[]): KeyRotationState {
        const active = keys.find((k) => k.retiredAt === null) ?? null;
        const retired = keys.filter((k) => k.retiredAt !== null);
        const lastRotated = retired.map((k) => k.retiredAt as string).sort().pop() ?? null;
        return {
          activeKeyId: active?.keyId ?? null,
          retiredKeyIds: retired.map((k) => k.keyId),
          lastRotatedAt: lastRotated,
        };
   }

   /** Is a given key id still usable for NEW signatures? Only the active key is. */
   export function isActiveKey(state: KeyRotationState, keyId: string): boolean {
        return state.activeKeyId === keyId;
   }


   /** A retired key can still VERIFY old signatures, just not sign new ones. */
   export function canVerifyWithKey(keys: SigningPublicKey[], keyId: string): boolean {
     return keys.some((k) => k.keyId === keyId);
   }

   import { createCipheriv, createDecipheriv, randomBytes, scryptSync } from 'node:crypto';
   import type { KmsProvider } from './kmsProvider';


   /**
    * LOCAL DEV KMS — NOT FOR PRODUCTION. Derives a master key from an env secret and
       * wraps data keys with AES-256-GCM. Use a real KMS in production. If
       * REMOTEDESK_KMS_DEV_SECRET is unset, it refuses to construct (fail closed)
       * rather than using a hardcoded key.
       */
   export class LocalDevKms implements KmsProvider {
     readonly masterKeyRef = 'local-dev-kms';
        private readonly master: Buffer;


        constructor(secret = process.env.REMOTEDESK_KMS_DEV_SECRET) {
          if (!secret) {
                throw new Error('LocalDevKms requires REMOTEDESK_KMS_DEV_SECRET (dev only). Use a real KMS in production.');
            }
            this.master = scryptSync(secret, 'remotedesk-kms-salt', 32);
        }


        async generateDataKey(): Promise<{ plaintext: Buffer; wrapped: string }> {
            const plaintext = randomBytes(32);
            return { plaintext, wrapped: this.wrap(plaintext) };
        }


        async unwrap(wrapped: string): Promise<Buffer> {
          const raw = Buffer.from(wrapped, 'base64');
            const iv = raw.subarray(0, 12);
            const tag = raw.subarray(12, 28);
            const ct = raw.subarray(28);
            const decipher = createDecipheriv('aes-256-gcm', this.master, iv);
            decipher.setAuthTag(tag);
            return Buffer.concat([decipher.update(ct), decipher.final()]);
        }


        private wrap(plaintext: Buffer): string {
          const iv = randomBytes(12);
            const cipher = createCipheriv('aes-256-gcm', this.master, iv);
            const ct = Buffer.concat([cipher.update(plaintext), cipher.final()]);
            const tag = cipher.getAuthTag();
            return Buffer.concat([iv, tag, ct]).toString('base64');
        }
   }


   /** Factory: swap to a real KMS provider here based on env. */
   export function resolveKmsProvider(): KmsProvider {
     // if (process.env.AWS_KMS_KEY_ID) return new AwsKmsProvider(...);        // production
        return new LocalDevKms();
   }

   import { createCipheriv, createDecipheriv, randomBytes } from 'node:crypto';
   import type { KmsProvider } from './kmsProvider';

   /**
    * Envelope encryption for signing private keys:
       *     - KMS generates a data key (wrapped at rest, plaintext in memory)
       *     - we AES-256-GCM encrypt the private key with the plaintext data key
       *     - we persist { wrappedDataKey, iv, tag, ciphertext } — NEVER the plaintext key
       */
   export interface EncryptedKeyMaterial {
     wrappedDataKey: string;
           iv: string;
           tag: string;
           ciphertext: string;
   }


   export async function encryptPrivateKey(kms: KmsProvider, privateKeyB64Url: string): Promise<EncryptedKeyMaterial> {
           const { plaintext: dataKey, wrapped } = await kms.generateDataKey();
           const iv = randomBytes(12);
           const cipher = createCipheriv('aes-256-gcm', dataKey, iv);
           const ct = Buffer.concat([cipher.update(Buffer.from(privateKeyB64Url, 'utf8')), cipher.final()]);
           const tag = cipher.getAuthTag();
           dataKey.fill(0); // zero the plaintext data key ASAP
     return { wrappedDataKey: wrapped, iv: iv.toString('base64'), tag: tag.toString('base64'), ciphertext:
   ct.toString('base64') };
   }


   export async function decryptPrivateKey(kms: KmsProvider, m: EncryptedKeyMaterial): Promise<string> {
     const dataKey = await kms.unwrap(m.wrappedDataKey);
           const decipher = createDecipheriv('aes-256-gcm', dataKey, Buffer.from(m.iv, 'base64'));
           decipher.setAuthTag(Buffer.from(m.tag, 'base64'));
           const pt = Buffer.concat([decipher.update(Buffer.from(m.ciphertext, 'base64')), decipher.final()]);
           dataKey.fill(0); // zero the plaintext data key ASAP
           return pt.toString('utf8');
   }

   /**
    * Signing-key rotation. Retires the active key + mints a new one, audited via the
    * Pack 25 trail. Old keys remain for verifying historical receipts.
    */
   export const keyRotationService = {
     async state(orgId: string): Promise<KeyRotationState> {
          const keys: SigningPublicKey[] = await (signingKeyRepository as any).listPublic(orgId);
          return deriveRotationState(keys);
        },


        async rotate(orgId: string, actorUserId: string | null): Promise<string> {
          const newKeyId = await (signingKeyRepository as any).rotate(orgId);
          await deviceAdminAudit.record({
            orgId, actorUserId, deviceId: 'signing-key',
             action: 'device.trust.changed', // or a dedicated 'signing.key.rotated' action
             metadata: { event: 'signing_key_rotated', newKeyId },
          });
          return newKeyId;
        },
   };

   'use client';


   import React, { useEffect, useState } from 'react';
   import type { KeyRotationState, SigningPublicKey } from '@remotedesk/shared';
   import { api } from '../../../src/api/apiClient';


   /**
    * Admin panel: active signing public key + rotation state + a Rotate button.
    * Never shows private material (the API can't return it).
    */
   export function SigningKeyPanel({ canManage }: { canManage: boolean }) {
        const [pub, setPub] = useState<SigningPublicKey | { keyId: null } | null>(null);
        const [state, setState] = useState<KeyRotationState | null>(null);
        const [busy, setBusy] = useState(false);


        const load = async () => {
          setPub(await api.get('/devices/receipts/signing-key'));
          if (canManage) setState(await api.get<KeyRotationState>('/devices/receipts/signing-key/state'));
        };
        useEffect(() => { void load(); }, []); // eslint-disable-line react-hooks/exhaustive-deps


        const rotate = async () => {
          setBusy(true);
         try { await api.post('/devices/receipts/signing-key/rotate', {}); await load(); }
         finally { setBusy(false); }
       };


       return (
         <section className="signing-key-panel">
           <h2>Receipt signing key</h2>
             <p className="mono">
               Active key: {pub && 'keyId' in pub && pub.keyId ? pub.keyId : 'none yet (sign a receipt to create one)'}
             </p>
             {state && (
               <p className="muted">
                 Retired keys: {state.retiredKeyIds.length}
                 {state.lastRotatedAt ? ` · last rotated ${new Date(state.lastRotatedAt).toLocaleString()}` : ''}
               </p>
             )}
             {canManage && (
               <button type="button" disabled={busy} onClick={rotate}>
                 {busy ? 'Rotating…' : 'Rotate signing key'}
               </button>
             )}
           <p className="muted">Old keys are retired, not deleted, so past receipts stay verifiable.</p>
         </section>
       );
   }
