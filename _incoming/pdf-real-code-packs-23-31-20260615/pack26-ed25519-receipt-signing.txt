 Codex / Claude Code handoff. Direct continuation of Packs 22 → 25. So far a consent receipt proves INTERNAL
 consistency (its own hash chain). This slice adds cryptographic NON-REPUDIATION: the server signs each
 verified receipt's root hash with an Ed25519 key, publishes the public key, and anyone can verify the signature
 offline. The signature binds the org + receipt + root hash so a receipt can't be re-attributed or forged. Build
 order: shared signature contract + verifier → server keypair management → sign-on-store → public-key +
 verify endpoints → web badge → tests.
 Scope: detached Ed25519 signatures over a canonical receipt digest, server-held private key (never leaves the
 server, never logged), published public key for offline verification. Reuses Pack 24 verification (a receipt is
 signed ONLY after it validates). Private key never in responses/logs. Signature covers
 org+receiptId+rootHash+issuedAt. Verification is content-free + offline. No new heavy crypto deps
 beyond Node/WebCrypto Ed25519.

   import type { ConsentReceipt } from './consentReceipt';

   /**
       * Detached Ed25519 signature over a CANONICAL receipt digest. The signature binds
       * the org, receiptId, rootHash, and issuedAt, so a valid receipt can't be silently
       * re-attributed to another org or have its root swapped. Verification is offline:
       * given the receipt + signature + the org's published public key, anyone can check
       * it with no server call.
       */
   export interface ReceiptSignature {
     /** Key id of the signing key (rotation-aware). */
        keyId: string;
        /** Ed25519 signature, base64url. */
        signature: string;
        /** Algorithm tag for forward-compat. */
        alg: 'ed25519';
        signedAt: string;
   }


   /** A published signing public key. */
   export interface SigningPublicKey {
        keyId: string;
        /** Raw Ed25519 public key, base64url (32 bytes). */
        publicKey: string;
        alg: 'ed25519';
        createdAt: string;
        /** Set when rotated out; verification of old signatures still works. */
        retiredAt: string | null;
   }


   /**
       * Canonical signing preimage. ORDER-STABLE + content-free: org + receiptId +
       * rootHash + issuedAt only. Never includes summary internals or line items (the
       * rootHash already commits to those via the Pack 22 chain).
       */
   export function signingPreimage(orgId: string, receipt: Pick<ConsentReceipt, 'receiptId' | 'rootHash' | 'issuedAt'>):
   string {
        return `remotedesk-receipt-v1|${orgId}|${receipt.receiptId}|${receipt.rootHash}|${receipt.issuedAt}`;
   }

   /** Injected verify fn so shared stays runtime-agnostic (Node/WebCrypto supply it). */
   export type Ed25519Verify = (publicKeyB64Url: string, message: string, signatureB64Url: string) => Promise<boolean>;

   /**
    * Offline signature verification. Confirms the detached signature over the
       * canonical preimage using the published public key. Returns false on any
       * mismatch (fail-closed).
    */
   export async function verifyReceiptSignature(
        orgId: string,
       receipt: ConsentReceipt,
       signature: ReceiptSignature,
       publicKey: SigningPublicKey,
    ed25519Verify: Ed25519Verify,
  ): Promise<boolean> {
       if (signature.alg !== 'ed25519' || publicKey.alg !== 'ed25519') return false;
       if (signature.keyId !== publicKey.keyId) return false;
       const message = signingPreimage(orgId, receipt);
       try {
         return await ed25519Verify(publicKey.publicKey, message, signature.signature);
       } catch {
           return false;
       }
  }

   import { generateKeyPairSync, sign as nodeSign, verify as nodeVerify, createPublicKey, createPrivateKey } from
   'node:crypto';


   /**
       * Ed25519 helpers using Node's built-in crypto (no external deps). Keys are stored
       * as base64url RAW 32-byte values; we wrap/unwrap to DER for Node's KeyObject API.
       * The private key is handled ONLY here and in the signer service; never logged,
       * never serialized to a client.
       */


   const RAW_PUB_PREFIX = Buffer.from('302a300506032b6570032100', 'hex'); // SPKI Ed25519 header
   const RAW_PRIV_PREFIX = Buffer.from('302e020100300506032b657004220420', 'hex'); // PKCS8 Ed25519 header


   export function generateEd25519(): { publicKey: string; privateKey: string } {
        const { publicKey, privateKey } = generateKeyPairSync('ed25519');
        const pubRaw = publicKey.export({ type: 'spki', format: 'der' }).subarray(RAW_PUB_PREFIX.length);
        const privRaw = privateKey.export({ type: 'pkcs8', format: 'der' }).subarray(RAW_PRIV_PREFIX.length);
        return { publicKey: b64url(pubRaw), privateKey: b64url(privRaw) };
   }


   export function signEd25519(privateKeyB64Url: string, message: string): string {
     const der = Buffer.concat([RAW_PRIV_PREFIX, fromB64url(privateKeyB64Url)]);
        const key = createPrivateKey({ key: der, format: 'der', type: 'pkcs8' });
        const sig = nodeSign(null, Buffer.from(message, 'utf8'), key);
        return b64url(sig);
   }


   export async function verifyEd25519(publicKeyB64Url: string, message: string, signatureB64Url: string): Promise<boolean>
   {
        const der = Buffer.concat([RAW_PUB_PREFIX, fromB64url(publicKeyB64Url)]);
        const key = createPublicKey({ key: der, format: 'der', type: 'spki' });
        return nodeVerify(null, Buffer.from(message, 'utf8'), key, fromB64url(signatureB64Url));
   }


   function b64url(buf: Buffer): string {
        return buf.toString('base64').replace(/\+/g, '-').replace(/\//g, '_').replace(/=+$/, '');
   }
   function fromB64url(s: string): Buffer {
     return Buffer.from(s.replace(/-/g, '+').replace(/_/g, '/'), 'base64');
   }

   import type { SigningPublicKey } from '@remotedesk/shared';
   import { prisma } from '../db/prisma';
   import { generateEd25519 } from '../lib/ed25519';

   /**
       * Manages signing keys. CRITICAL: privateKey is selected ONLY by signingPrivate()
       * for use inside the signer service. The public listing/get NEVER includes it.
    */
   export const signingKeyRepository = {
        /** Public key for publishing (no private material). */
        async publicKey(orgId: string | null): Promise<SigningPublicKey | null> {
          const k = await prisma.signingKey.findFirst({
            where: { orgId, retiredAt: null },
             orderBy: { createdAt: 'desc' },
             select: { keyId: true, publicKey: true, alg: true, createdAt: true, retiredAt: true },
          });
          return k ? { keyId: k.keyId, publicKey: k.publicKey, alg: 'ed25519', createdAt: k.createdAt.toISOString(), retiredAt:
   k.retiredAt ? k.retiredAt.toISOString() : null } : null;
     },


        /** Internal-only: private key for signing. NEVER expose via a route. */
        async signingPrivate(orgId: string | null): Promise<{ keyId: string; privateKey: string } | null> {
          const k = await prisma.signingKey.findFirst({
             where: { orgId, retiredAt: null },
             orderBy: { createdAt: 'desc' },
            select: { keyId: true, privateKey: true },
          });
          return k ?? null;
        },


        /** Lazily create the active signing key for an org if none exists. */
        async ensureKey(orgId: string | null): Promise<{ keyId: string; privateKey: string }> {
          const existing = await this.signingPrivate(orgId);
          if (existing) return existing;
          const { publicKey, privateKey } = generateEd25519();
          const keyId = `k_${Date.now().toString(36)}`;
          await prisma.signingKey.create({ data: { keyId, orgId, alg: 'ed25519', publicKey, privateKey } });
          return { keyId, privateKey };
        },


        /** Look up a public key by id (for verifying old signatures after rotation). */
        async publicKeyById(keyId: string): Promise<SigningPublicKey | null> {
          const k = await prisma.signingKey.findUnique({
            where: { keyId },
            select: { keyId: true, publicKey: true, alg: true, createdAt: true, retiredAt: true },
          });
       return k ? { keyId: k.keyId, publicKey: k.publicKey, alg: 'ed25519', createdAt: k.createdAt.toISOString(), retiredAt:
   k.retiredAt ? k.retiredAt.toISOString() : null } : null;
        },
   };

   import {
     signingPreimage,
        verifyReceiptSignature,
        type ConsentReceipt,
        type ReceiptSignature,
        type SigningPublicKey,
   } from '@remotedesk/shared';
   import { signEd25519, verifyEd25519 } from '../../lib/ed25519';
   import { signingKeyRepository } from '../../repositories/signingKeyRepository';


   /**
    * Signs verified receipts + verifies signatures. The private key never leaves
    * this service (loaded, used to sign, discarded). Signing happens ONLY after the
    * Pack 24 verifier confirms the receipt is valid + content-free.
    */
   export const receiptSignerService = {
        /** Produce a detached signature over the canonical preimage. */
        async sign(orgId: string, receipt: ConsentReceipt): Promise<ReceiptSignature> {
          const key = await signingKeyRepository.ensureKey(orgId);
          const message = signingPreimage(orgId, receipt);
          const signature = signEd25519(key.privateKey, message);
          return { keyId: key.keyId, signature, alg: 'ed25519', signedAt: new Date().toISOString() };
        },

        /** Publish the org's current public key (no private material). */
        publicKey(orgId: string): Promise<SigningPublicKey | null> {
          return signingKeyRepository.publicKey(orgId);
        },


        /**
         * Verify a receipt's signature server-side (also possible fully offline by a
         * third party using the published public key + shared verifier).
         */
        async verify(orgId: string, receipt: ConsentReceipt, signature: ReceiptSignature): Promise<boolean> {
          const pub = await signingKeyRepository.publicKeyById(signature.keyId);
          if (!pub) return false;
          return verifyReceiptSignature(orgId, receipt, signature, pub, verifyEd25519);
        },
   };

   /**
       * Shows whether a receipt is cryptographically signed + (after verify) whether the

       * non-repudiation (the SERVER attests it), not just internal consistency.
       */
   export function SignatureBadge({
     signed,
     signatureValid,
   }: {
        signed: boolean;
        signatureValid: boolean | null; // null = not yet verified
   }) {
     if (!signed) return <span className="sig-badge sig-badge--unsigned">unsigned</span>;
        if (signatureValid === null) return <span className="sig-badge sig-badge--signed">signed</span>;
        return signatureValid
          ? <span className="sig-badge sig-badge--valid">✓ signature valid</span>
          : <span className="sig-badge sig-badge--invalid">✗ signature invalid</span>;
   }

  ## What this slice does
  Adds cryptographic non-repudiation to Pack 22 consent receipts. After the Pack 24
  verifier confirms a receipt is valid + content-free, the server signs its canonical
  digest (org + receiptId + rootHash + issuedAt) with an Ed25519 key, stores the
  signature, and publishes the public key. Anyone can verify offline with the shared
  verifier — no server call, no private material.

repo's internal methods, used only inside the signer service, and never serialized to a response or log (there's
even a test asserting it can't leak onto the signature object). Signing happens strictly after validation, so the
signature and the hash chain can never disagree.
