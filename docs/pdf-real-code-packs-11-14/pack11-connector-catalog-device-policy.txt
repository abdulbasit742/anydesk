  import type { ConnectorDefinition, ConnectorStatus } from './types';
  import { getConnectorDefinition } from './catalog';


  /**
   * Pure safety guards shared by server and client so install rules never drift.
      * Pack 11 has no OAuth/secrets — these guards enforce that invariant.
      */


  /** Field names that must never appear on a connector definition payload. */
  export const FORBIDDEN_SECRET_KEYS: readonly string[] = [
    'clientSecret',
       'client_secret',
       'apiKey',
       'api_key',
       'token',
       'accessToken',
       'refreshToken',
       'password',
       'privateKey',
  ];


  /** Throws if a (possibly untrusted) object carries secret-looking fields. */
  export function assertNoSecretFields(obj: Record<string, unknown>): void {
       for (const key of Object.keys(obj)) {
          if (FORBIDDEN_SECRET_KEYS.includes(key)) {
              throw new Error(`Connector payload must not contain secret field "${key}"`);
          }
      }
  }


  /** A connector can be installed only if it exists and is not coming_soon. */
  export function isInstallable(key: string): boolean {
    const def = getConnectorDefinition(key);
      return !!def && def.availability !== 'coming_soon';
  }

  /** Compute the effective UI status from availability + install state. */
  export function resolveStatus(
    def: Pick<ConnectorDefinition, 'availability'>,
    installed: boolean,
  ): ConnectorStatus {
      if (def.availability === 'coming_soon') return 'coming_soon';
      return installed ? 'installed' : 'available';
  }


  /** Normalize/validate a connector key into its canonical slug, or null. */
  export function normalizeConnectorKey(raw: string): string | null {
      const key = raw.trim().toLowerCase();
      return getConnectorDefinition(key) ? key : null;
  }

  /**
   * Device policy contract shared by the API (source of truth) and the desktop
    * client (enforcer). Enforcement fails CLOSED: when policy is unknown, every
    * sensitive capability is denied.
    */
export interface DevicePolicy {
  deviceId: string;
     /** Operator may send mouse/keyboard to this device. */
     allowRemoteInput: boolean;
     /** Clipboard sync (copy/paste) permitted. */
     allowClipboard: boolean;
     /** File transfer to/from this device permitted. */
     allowFileTransfer: boolean;
     /** Remote audio capture permitted. */
     allowAudio: boolean;
     /** Hard session cap in minutes; null = no cap. */
     maxSessionMinutes: number | null;
     /** Monotonic version for staleness checks. */
     version: number;
     updatedAt: string;
}


/** The capabilities the desktop session gates on. */
export type PolicyCapability =
  | 'remoteInput'
     | 'clipboard'
     | 'fileTransfer'
     | 'audio';


/**
 * The fail-closed policy. Used whenever the API is unreachable, the payload is
    * malformed, or no policy has ever been fetched. Everything denied.
    */
export const FAIL_CLOSED_POLICY: Omit<DevicePolicy, 'deviceId' | 'updatedAt'> = {
  allowRemoteInput: false,
     allowClipboard: false,
     allowFileTransfer: false,
     allowAudio: false,
     maxSessionMinutes: 0,
     version: 0,
};


/** Build a concrete fail-closed policy for a device. */
export function failClosedPolicy(deviceId: string): DevicePolicy {
  return {
       deviceId,
       ...FAIL_CLOSED_POLICY,
       updatedAt: new Date(0).toISOString(),
     };
}


/** Map a capability to its policy flag. Pure, used by client + tests. */
export function isCapabilityAllowed(
     policy: DevicePolicy,
     capability: PolicyCapability,
): boolean {
  switch (capability) {
       case 'remoteInput':
         return policy.allowRemoteInput;
          case 'clipboard':
            return policy.allowClipboard;
          case 'fileTransfer':
            return policy.allowFileTransfer;
          case 'audio':
            return policy.allowAudio;
          default:
           return false;
      }
  }


  /** Runtime guard: is this an acceptably-shaped DevicePolicy? */
  export function isDevicePolicy(value: unknown): value is DevicePolicy {
      if (!value || typeof value !== 'object') return false;
      const p = value as Record<string, unknown>;
      return (
        typeof p.deviceId === 'string' &&
          typeof p.allowRemoteInput === 'boolean' &&
          typeof p.allowClipboard === 'boolean' &&
          typeof p.allowFileTransfer === 'boolean' &&
          typeof p.allowAudio === 'boolean' &&
          typeof p.version === 'number'
      );
  }

  import {
       DEFAULT_CONNECTORS,
       getConnectorDefinition,
       resolveStatus,
       type ConnectorCatalogEntry,
    type ConnectorDefinition,
  } from '@remotedesk/shared';


  /**
      * Pure catalog assembly: merge the code-authored definitions with an org's
      * installation map into the read-only catalog the API serves. No DB access here
      * so it stays trivially testable.
      */
  export function buildCatalog(
    installed: ReadonlyMap<string, { installedAt: Date }>,
  ): ConnectorCatalogEntry[] {
    return DEFAULT_CONNECTORS.map((def) => toEntry(def, installed.get(def.key)));
  }

  export function toEntry(
    def: ConnectorDefinition,
    install?: { installedAt: Date },
  ): ConnectorCatalogEntry {
      const isInstalled = !!install;
      return {
        ...def,
        installed: isInstalled,
        installedAt: install ? install.installedAt.toISOString() : null,
        status: resolveStatus(def, isInstalled),
      };
  }

  /** Re-export for callers that only need a single definition. */
  export { getConnectorDefinition };

  export const connectorRepository = {

      async installationsForOrg(
        orgId: string,
      ): Promise<Map<string, { installedAt: Date }>> {
        const rows = await prisma.connectorInstallation.findMany({ where: { orgId } });
        return new Map(rows.map((r) => [r.connectorKey, { installedAt: r.installedAt }]));
      },


      async isInstalled(orgId: string, connectorKey: string): Promise<boolean> {
        const row = await prisma.connectorInstallation.findUnique({
          where: { orgId_connectorKey: { orgId, connectorKey } },
        });
        return !!row;
      },


      async install(
        orgId: string,
        connectorKey: string,
        installedById: string,
        label?: string,
      ): Promise<void> {
        await prisma.connectorInstallation.create({
          data: { orgId, connectorKey, installedById, label: label ?? null },
        });
      },

      async uninstall(orgId: string, connectorKey: string): Promise<number> {
        const { count } = await prisma.connectorInstallation.deleteMany({
          where: { orgId, connectorKey },
        });
        return count;
      },


      async writeAudit(
        orgId: string,
        connectorKey: string,
         action: ConnectorAuditAction,

  import {
    AppError,
       ErrorCodes,
       getConnectorDefinition,
       isInstallable,
       type ConnectorCatalogEntry,
  } from '@remotedesk/shared';
  import { buildCatalog, toEntry } from '../../lib/connectorCatalog';
  import { connectorRepository } from '../../repositories/connectorRepository';
  import { connectorAudit } from './connectorAudit';


  export const connectorService = {
       /** Read-only catalog for an org. */
       async catalog(orgId: string): Promise<ConnectorCatalogEntry[]> {
         const installed = await connectorRepository.installationsForOrg(orgId);
         return buildCatalog(installed);
       },
       /** Install placeholder: records the installation + audit. No provider calls. */
       async install(
         orgId: string,
         connectorKey: string,
         actorUserId: string,
         label?: string,
       ): Promise<ConnectorCatalogEntry> {
         const def = getConnectorDefinition(connectorKey);
         if (!def) throw new AppError(ErrorCodes.CONNECTOR_NOT_FOUND);
         if (!isInstallable(connectorKey)) {
           throw new AppError(ErrorCodes.CONNECTOR_NOT_INSTALLABLE);
         }
         if (await connectorRepository.isInstalled(orgId, connectorKey)) {
             throw new AppError(ErrorCodes.CONNECTOR_ALREADY_INSTALLED);
         }
         await connectorRepository.install(orgId, connectorKey, actorUserId, label);
         await connectorAudit.record(orgId, connectorKey, 'connector.installed', actorUserId);
         return toEntry(def, { installedAt: new Date() });
       },


       /** Uninstall placeholder: removes the installation + audit. */
       async uninstall(
         orgId: string,
         connectorKey: string,
         actorUserId: string,
       ): Promise<ConnectorCatalogEntry> {
         const def = getConnectorDefinition(connectorKey);
         if (!def) throw new AppError(ErrorCodes.CONNECTOR_NOT_FOUND);
         const removed = await connectorRepository.uninstall(orgId, connectorKey);
         if (removed === 0) throw new AppError(ErrorCodes.CONNECTOR_NOT_INSTALLED);
         await connectorAudit.record(orgId, connectorKey, 'connector.uninstalled', actorUserId);
         return toEntry(def, undefined);
       },
  };

    (fn: (req: any, res: any) => Promise<void>) =>
       (req: any, res: any, next: any) =>
         fn(req, res).catch(next);


  /**
   * /api/connectors
   *   GET    /                   -> read-only catalog for the caller's org
   *        POST /:key/install    -> install placeholder (audited)
   *        DELETE /:key          -> uninstall placeholder (audited)
   *
      * orgContext resolves :orgId from the request (header/body) + the caller's role.
   */
  export const connectorRoutes = Router()
       .get('/', requireAuth, orgContext, h(async (req, res) => {
         res.json(await connectorService.catalog((req as OrgRequest).orgId));
       }))
       .post('/:key/install', requireAuth, orgContext, h(async (req, res) => {
           const { key } = connectorKeyParamSchema.parse(req.params);
           const { label } = installConnectorSchema.parse(req.body ?? {});
           const r = req as OrgRequest & AuthedRequest;
           res.status(201).json(
             await connectorService.install(r.orgId, key, r.userId, label),
           );
       }))
       .delete('/:key', requireAuth, orgContext, h(async (req, res) => {
           const { key } = connectorKeyParamSchema.parse(req.params);
           const r = req as OrgRequest & AuthedRequest;
         res.json(await connectorService.uninstall(r.orgId, key, r.userId));
       }));

  import { DEFAULT_CONNECTORS } from '@remotedesk/shared';

  /**
      * Idempotent seed. Upserts the code-authored catalog into ConnectorDefinition.
      * Safe to run on every boot/deploy. Call from your existing seed entrypoint:
      *     await seedConnectors();
      */
  export async function seedConnectors(): Promise<void> {
    for (const def of DEFAULT_CONNECTORS) {
           await prisma.connectorDefinition.upsert({
             where: { key: def.key },
             create: {
               key: def.key,
                name: def.name,
                category: def.category,
                description: def.description,
                availability: def.availability,
                capabilities: [...def.capabilities],
                docsUrl: def.docsUrl ?? null,
             },
             update: {
                name: def.name,
                category: def.category,
                description: def.description,
                availability: def.availability,
                capabilities: [...def.capabilities],
                docsUrl: def.docsUrl ?? null,
             },
           });
       }
  }


  // Allow running standalone: `tsx src/db/seed/connectors.seed.ts`
  if (require.main === module) {
       seedConnectors()
        .then(() => {
          // eslint-disable-next-line no-console
          console.log(`Seeded ${DEFAULT_CONNECTORS.length} connectors`);
          return prisma.$disconnect();

  import {
    failClosedPolicy,
    isCapabilityAllowed,
    isDevicePolicy,
    type DevicePolicy,
    type PolicyCapability,
  } from '@remotedesk/shared';


  /**
   * In-memory policy store for the desktop client. Starts FAIL-CLOSED for the
   * device and only relaxes once a valid policy is fetched. Any malformed payload
   * is rejected and the store stays/returns to fail-closed.
   */
  export class DevicePolicyStore {
    private policy: DevicePolicy;


    constructor(private readonly deviceId: string) {
        this.policy = failClosedPolicy(deviceId);
    }

    get current(): DevicePolicy {
        return this.policy;
    }


    /** True only if a real (non-fail-closed) policy has been loaded. */
    get isHydrated(): boolean {
      return this.policy.version > 0;
    }


    /**
     * Accept a fetched payload. Returns true if applied. On any invalid/stale
     * payload we DROP to fail-closed (do not keep a possibly-revoked allow state
     * around longer than necessary).
     */
    apply(payload: unknown): boolean {
        if (!isDevicePolicy(payload) || payload.deviceId !== this.deviceId) {
          this.failClosed();
            return false;
        }
        // Ignore strictly older versions, but never let a stale fetch widen access.
           if (payload.version < this.policy.version) return false;
           this.policy = payload;
           return true;
       }


       /** Force everything denied (called on fetch failure / API unavailable). */
       failClosed(): void {
           this.policy = failClosedPolicy(this.deviceId);
       }

       allows(capability: PolicyCapability): boolean {
           return isCapabilityAllowed(this.policy, capability);
       }
  }

  import { type DevicePolicyStore } from './devicePolicy';


  /** Transport injected so tests don't touch the network. */
  export type PolicyFetch = (deviceId: string) => Promise<unknown>;


  export interface PollerOptions {
    intervalMs?: number;
       /** Called whenever the effective policy changes (e.g. to re-gate the UI). */
       onChange?: () => void;
  }


  /**
   * Polls the API for the device policy on an interval. On ANY failure (network,
      * non-2xx, malformed body) it forces the store fail-closed, so a device that
      * loses contact with the control plane immediately loses sensitive capabilities.
   */
  export class DevicePolicyPoller {
       private timer: ReturnType<typeof setInterval> | null = null;
       private readonly intervalMs: number;


       constructor(
           private readonly deviceId: string,
           private readonly store: DevicePolicyStore,
           private readonly fetchPolicy: PolicyFetch,
           private readonly opts: PollerOptions = {},
       ) {
         this.intervalMs = opts.intervalMs ?? 30_000;
       }


       async start(): Promise<void> {
         await this.tick();
           this.timer = setInterval(() => void this.tick(), this.intervalMs);
       }


       stop(): void {
           if (this.timer) clearInterval(this.timer);
           this.timer = null;
       }


       /** One poll cycle. Exposed for tests + manual refresh. */
       async tick(): Promise<void> {
           const before = this.store.current.version;
           try {
            const payload = await this.fetchPolicy(this.deviceId);
            this.store.apply(payload);
           } catch {
             this.store.failClosed();
           }
           if (this.store.current.version !== before) this.opts.onChange?.();
       }
  }


  /** Default transport over fetch. Throws on non-2xx so the poller fails closed. */
  export function httpPolicyFetch(baseUrl: string, authToken: string): PolicyFetch {
    return async (deviceId: string) => {
           const res = await fetch(`${baseUrl}/api/devices/${deviceId}/policy`, {
             headers: { authorization: `Bearer ${authToken}` },
           });
           if (!res.ok) throw new Error(`policy fetch failed: ${res.status}`);
           return res.json();
       };
  }

  | slack | Slack | chat | available |

  | jira | Jira | ticketing | available |
  | zendesk | Zendesk | ticketing | available |
  | s3 | Amazon S3 | storage | available |
  | siem-webhook | SIEM Webhook | security | coming soon |


  ## API
  - `GET /api/connectors` — catalog for the caller's org (definition + install state).
  - `POST /api/connectors/:key/install` — install placeholder; writes a `connector.installed` audit event.
  - `DELETE /api/connectors/:key` — uninstall placeholder; writes a `connector.uninstalled` audit event.

  `status` is derived: `coming_soon` always wins; otherwise `installed` vs
  `available`. The same `resolveStatus` helper runs on server and client.

  # Device Policy Enforcement (Desktop)


  The desktop client enforces a per-device policy over the three sensitive session
  capabilities — **remote input, clipboard sync, file transfer** (plus audio) —
  and **fails closed**.


  ## Fail-closed by default
  `DevicePolicyStore` starts at `failClosedPolicy(deviceId)`: every capability
  denied, `version = 0`. Capabilities are only enabled after a **valid** policy is
  fetched. Any of these drop the device back to fully denied:
  - API unreachable / non-2xx / network error
  - malformed policy payload
  - a policy whose `deviceId` doesn't match this device


  ## Components
  - `services/devicePolicy.ts` — in-memory `DevicePolicyStore` (apply/failClosed/allows).
  - `services/devicePolicyPoller.ts` — interval poller; on **any** fetch failure it forces fail-closed. Transport is
  injected (`httpPolicyFetch` for prod, mocks in tests).
  - `services/devicePolicyGuards.ts` — pure gate functions the session pipeline calls before sending input, syncing
  clipboard, or transferring files.
  ## Wiring
  `session/sessionSetup.ts` (patch) constructs the store + guards + poller during
  session bootstrap, gates the existing input/clipboard/file-transfer handlers on
  the guards, and stops the poller on teardown.


  ## Contract
  Policy shape is `DevicePolicy` in `packages/shared/src/policy/devicePolicy.ts`.
  The poller expects `GET /api/devices/:deviceId/policy` to return that shape.
  `version` is monotonic; a strictly older fetch never widens access.


  ## Why client-side gating
  The desktop client is the last line before the OS performs the action, so gating
  here guarantees a revoked or unreachable policy stops input/clipboard/file
  transfer even if a server-side check is bypassed. Server-side enforcement remains
  a follow-up.
